renderUI()

Render the search modal UI.

renderUI(
    placeholder: string,
    currentMode: 'regular' | 'opaque' | 'high-contrast',
    onModeChange: (mode: string) => void
): void {
  this.modeChangeCallback = onModeChange;
  const host = document.createElement('div');
  host.className = 'reef-host is-hidden';
  
  // Defer to DOM ready if called from head before body exists
  if (document.body) {
    document.body.appendChild(host);
  } else {
    const appendHost = () => {
      if (document.body) {
        document.body.appendChild(host);
      }
    };
    document.addEventListener('DOMContentLoaded', appendHost);
  }
  
  this.host = host;
  this.root = host.attachShadow({ mode: 'open' });
  const shadow = this.root;

  shadow.innerHTML = `
    <style>
      /* CSS styles here */
    </style>
    <div class="panel">
      <div class="input-row">
        <svg class="icon" viewBox="0 0 24 24">...</svg>
        <input class="input" type="text" placeholder="${placeholder}" />
        <button class="settings-toggle-btn"></button>
      </div>
      <div class="tabs-row">
        <button class="tab-chip" data-cat="all">All</button>
        <button class="tab-chip" data-cat="pages">Pages</button>
        <button class="tab-chip" data-cat="actions">Actions</button>
        <button class="tab-chip" data-cat="files">Files</button>
        <button class="tab-chip" data-cat="links">Links</button>
      </div>
      <div class="results-list"></div>
      <div class="settings-panel">...</div>
    </div>
  `;
  
  // Set up references
  this.input = shadow.querySelector('.input');
  this.resultsList = shadow.querySelector('.results-list');
  
  // Apply mode
  this.applyMode(currentMode);
  
  // Set up focus trap
  this.setupFocusTrap();
  
  // Update settings UI
  this.updateSettingsUI();
}

// - Input row with search icon, input field, and settings button
// - Category tabs for filtering results
// - Results list area
// - Settings panel (hidden by default)

renderResults(
  query: string,
  results: IndexRecord[],
  selectedIndex: number,
  onHover: (index: number) => void,
  onClick: (event: MouseEvent, index: number) => void
): void {
  const list = this.resultsList;
  if (!list) return;

  if (!results.length) {
    const suggestion = findClosestWord(query, this.index);
    if (suggestion) {
      list.innerHTML = `<div class="empty">No sections match "${this.escapeHtml(query)}". Did you mean <strong>${this.escapeHtml(suggestion)}</strong>?</div>`;
    } else {
      list.innerHTML = `<div class="empty">No sections match "${this.escapeHtml(query)}"</div>`;
    }
    const countEl = this.root?.querySelector('#count');
    if (countEl) countEl.textContent = '0 results';
    return;
  }

  // Render each result
  list.innerHTML = results.map((result, index) => `
    <div class="result ${index === selectedIndex ? 'selected' : ''}" data-index="${index}">
      <div class="result-type">${getResultTypeIcon(result.type)}</div>
      <div class="result-content">
        <div class="result-type-label">${getResultTypeLabel(result.type)}</div>
        <div class="heading">${highlight(result.headingText, query)}</div>
        <div class="breadcrumb">${this.escapeHtml(result.breadcrumb)}</div>
        <div class="snippet">${getSnippet(result.bodyText, query, 120)}</div>
        ${result.destructive ? '<div class="destructive-badge">Destructive</div>' : ''}
      </div>
    </div>
  `).join('');

  // Set up event listeners
  this.setupResultEventListeners(onHover, onClick);
  
  // Update count
  const countEl = this.root?.querySelector('#count');
  if (countEl) countEl.textContent = `${results.length} results`;
}

Focus Management

// setupFocusTrap() - Traps focus within the modal
setupFocusTrap(): void {
  const host = this.host;
  if (!host) return;

  const focusableSelector = 'input, button, [tabindex]:not([tabindex="-1"])';
  const focusable = Array.from(host.querySelectorAll<HTMLElement>(focusableSelector));
  this.focusableElements = focusable;

  host.addEventListener('keydown', (e) => {
    if (e.key === 'Tab') {
      const first = focusable[0];
      const last = focusable[focusable.length - 1];
      
      if (e.shiftKey && document.activeElement === first) {
        e.preventDefault();
        last.focus();
      } else if (!e.shiftKey && document.activeElement === last) {
        e.preventDefault();
        first.focus();
      }
    }
  });
}

// scrollSelectedIntoView() - Scrolls the selected result into view
scrollSelectedIntoView(index: number): void {
  const list = this.resultsList;
  if (!list) return;

  const selected = list.querySelector(`[data-index="${index}"]`);
  if (selected) {
    selected.scrollIntoView({ block: 'nearest', behavior: 'smooth' });
  }
}

escapeHtml()

Escape HTML special characters to prevent XSS.

export function escapeHtml(s: string): string {
  const map: Record<string, string> = { 
    '&': '&amp;', 
    '<': '&lt;', 
    '>': '&gt;', 
    '"': '&quot;', 
    "'": '''
  };
  let result = '';
  for (let i = 0; i < s.length; i++) {
    result += map[s[i]] ?? s[i];
  }
  return result;
}

// Usage
const safeText = escapeHtml(userInput);
// Converts: <script>alert('xss')</script>
// To: &lt;script&gt;alert('xss')&lt;/script&gt;

// Used throughout the UI rendering to prevent XSS attacks

highlight()

Highlight query matches in text by wrapping them in <mark> tags. Case-insensitive matching.

export function highlight(text: string, query: string): string {
  if (!query.trim()) return escapeHtml(text);
  const lowerText = text.toLowerCase();
  const lowerQuery = query.toLowerCase();
  const idx = lowerText.indexOf(lowerQuery);
  if (idx === -1) return escapeHtml(text);
  const before = escapeHtml(text.slice(0, idx));
  const match = escapeHtml(text.slice(idx, idx + query.length));
  const after = escapeHtml(text.slice(idx + query.length));
  return `${before}<mark>${match}</mark>${after}`;
}

// Usage examples
const highlighted = highlight('Installation Guide', 'install');
// Result: <mark>Install</mark>ation Guide

// Case-insensitive: highlight('User Settings', 'SET') 
// Result: User <mark>Set</mark>tings

// No match: highlight('Hello World', 'foo')
// Result: Hello World (no mark tags added)

// Empty query: highlight('Hello', '')
// Result: Hello (escaped but no highlighting)

// Multiple matches: Only the FIRST match is highlighted
// highlight('foo bar foo baz', 'foo') 
// Result: <mark>foo</mark> bar foo baz

getSnippet()

Extract a snippet of text around the query match, with context on both sides.

export function getSnippet(text: string, query: string): string {
  if (!query.trim()) return text.slice(0, 90) + '\u2026';
  const lowerText = text.toLowerCase();
  const lowerQuery = query.toLowerCase();
  const idx = lowerText.indexOf(lowerQuery);
  if (idx === -1) return text.slice(0, 90) + '\u2026';
  const start = Math.max(0, idx - 30);
  const end = Math.min(text.length, idx + query.length + 40);
  return (start > 0 ? '\u2026' : '') + text.slice(start, end) + (end < text.length ? '\u2026' : '');
}

// Usage examples
getSnippet('This is a long text about installation guide', 'installation');
// Result: "...long text about installation gu..."
//         ^30 chars before      ^40 chars after

getSnippet('Short text', 'text');
// Result: "Short text" (no ellipsis if full text fits)

getSnippet('No match here', 'missing');
// Result: "No match he..." (first 90 chars + ellipsis)

export function getResultTypeIcon(type: string): string {
  switch (type) {
    case 'section': return '<svg>...</svg>'; // Document icon
    case 'action':  return '<svg>...</svg>'; // Play/polygon icon
    case 'field':   return '<svg>...</svg>'; // Input/circle icon
    case 'link':    return '<svg>...</svg>'; // Link icon

}

getResultTypeLabel()

export function getResultTypeLabel(type: string): string {
  switch (type) {
    case 'section':  return 'Section';

    case 'field':    return 'Field';
    case 'link':     return 'Link';
    case 'file':     return 'File';
    case 'media':    return 'Media';

Complete Search Integration

Full example showing UI integration with search functionality.

// Initialize
const ui = new UIRenderer();
const inspector = new VisualInspector();

// Render UI
ui.renderUI('Search...', 'regular', handleModeChange);

// Wire up to search results
function displayResults(query, results) {
  const selectedIndex = 0;
  
  ui.renderResults(
    query,
    results,
    selectedIndex,
    (index) => { /* hover callback */ },
    (event, index) => { /* click callback */ }
  );
  
  // Update diagnostics
  ui.updateDiagnostics({
    pages: results.filter(r => r.type === 'section').length,
    interactive: results.filter(r => r.type === 'action' || r.type === 'field').length,
    files: results.filter(r => r.type === 'file').length
  });
}

// Toggle modal visibility
function openModal() {
  ui.setIsOpen(true);
  ui.applyAriaHidden();
  document.addEventListener('keydown', handleKeydown);
}

function closeModal() {
  ui.setIsOpen(false);
  ui.restoreBodyAriaHidden();
  document.removeEventListener('keydown', handleKeydown);
}

// Keyboard navigation
function handleKeydown(event) {
  if (event.key === 'Escape') {
    closeModal();
  }
}
