Overview

The actions module handles action execution, navigation, and deferred action management. The ActionExecutor class is responsible for executing actions when users select search results. 

/**
 * @file Action execution and navigation logic.
 * Handles navigating to sections, executing actions on elements, and deferred actions.
 */

import type { IndexRecord } from '../types.js';
import { escapeHtml } from '../ui/ui-helpers.js';
import { findClosestWord } from '../search-index.js';

export class ActionExecutor {
  private deferredScrollHandler: () => void = () => {};
  private deferredActionHandler: () => void = () => {};
  
  // ... methods
}

// Re-export from index.ts
export { ActionExecutor } from './action-executor.js';

// Usage
import { ActionExecutor } from './src/actions/index.js';

const executor = new ActionExecutor();
executor.executeAction(result, 'execute', showToast);

Navigation

executeAction()

The main entry point for executing actions based on result type.

executeAction(result: IndexRecord, actionsMode?: string, showToast?: (msg: string) => void): void {
  switch (result.type) {
    case 'action':
      this.executeActionResult(result, actionsMode, showToast);
      break;
    case 'field':
      this.focusField(result);
      break;
    case 'link':
    case 'file':
    case 'media':
    case 'structured':
      window.location.href = result.url;
      break;
    case 'section':
      this.navigateToSection(result, () => {});
      break;
  }
}

// Result type determines behavior:
// - 'action': Execute the action (click) or navigate first if on different page
// - 'field': Focus the field and select its content
// - 'link', 'file', 'media', 'structured': Navigate to the URL
// - 'section': Navigate to the section or scroll to it

// Usage in ReefSearch:
this.executor.executeAction(result, this.config.actionsMode, (msg) => this.ui.showToast(msg));

navigateToSection()

Navigate to a section, with special handling for same-page sections.

navigateToSection(result: IndexRecord, closeCallback: () => void): void {
  const currentUrl = window.location.href.split('#')[0];
  const targetUrl = result.url.split('#')[0];
  const isSamePage = currentUrl === targetUrl;

  // 1. Try to find element by selector
  if (result.selector) {
    const element = document.querySelector(result.selector);
    if (element) {
      this.highlightAndScrollTo(element as HTMLElement, closeCallback);
      return;
    }
  }

  // 2. If same page, try to find heading by text
  if (isSamePage) {
    const heading = this.findHeadingElementByText(result.headingText);
    if (heading) {
      this.highlightAndScrollTo(heading, closeCallback);
      return;
    }
  }

  // 3. If not found, set up deferred scroll and navigate
  this.setupDeferredScroll(result);
  closeCallback();
  window.location.href = result.url;
}

// This method tries multiple strategies to find the target:
// 1. First, try to find by CSS selector
// 2. Then, try to find by heading text
// 3. Finally, navigate to the page and set up deferred scroll
// This ensures smooth UX even if the element isn't immediately available

highlightAndScrollTo()

Visual feedback for selected results.

private highlightAndScrollTo(element: HTMLElement, closeCallback: () => void): void {
  closeCallback();
  element.scrollIntoView({ behavior: 'smooth', block: 'center' });
  element.style.boxShadow = '0 0 0 3px rgba(67, 217, 200, 0.5)';
  element.style.borderRadius = '4px';
  setTimeout(() => {
    element.style.boxShadow = '';
  }, 2000);
}

// This provides visual feedback:
// 1. Scrolls the element smoothly into the center of the viewport
// 2. Adds a temporary highlight (box shadow) to the element
// 3. Removes the highlight after 2 seconds
// The highlight uses Reef's primary color (teal/cyan)

focusField()

Focus a form field and select its content.

private focusField(result: IndexRecord): void {
  if (!result.selector) return;
  try {
    const element = document.querySelector(result.selector);
    if (element) {
      (element as HTMLElement).focus();
      if (element instanceof HTMLInputElement || element instanceof HTMLTextAreaElement) {
        element.select();
      }
    }
  } catch (error) {
    console.error('Failed to focus field:', error);
  }
}

// For field results:
// 1. Find the element by selector
// 2. Focus the element
// 3. If it's an input or textarea, select all its content
// This makes it easy for users to immediately start typing

Deferred Actions

Deferred actions handle the case where a user selects an action that targets a different page. The action is stored and executed when the target page loads.

setupDeferredScroll()

Set up a deferred scroll to a section.

private setupDeferredScroll(result: IndexRecord): void {
  const target = {
    headingText: result.headingText,
    selector: result.selector,
  };
  try {
    sessionStorage.setItem('reef-deferred-scroll', JSON.stringify(target));
  } catch (error) {
    console.error('[reef] Failed to store deferred scroll target:', error);
  }
}

// Stores the target in sessionStorage for retrieval on the next page load
// Uses sessionStorage (not localStorage) because it's per-tab

handleDeferredScroll()

Handle the deferred scroll when the target page loads.

handleDeferredScroll(): void {
  const raw = sessionStorage.getItem('reef-deferred-scroll');
  if (!raw) return;
  sessionStorage.removeItem('reef-deferred-scroll');

  let target: { headingText: string; selector?: string } | null = null;
  try {
    target = JSON.parse(raw);
  } catch (error) {
    console.error('[reef] Failed to parse deferred scroll target:', error);
    return;
  }
  if (!target || !target.headingText) return;
  const deferredTarget = target;

  const findTarget = (): HTMLElement | null => {
    if (deferredTarget.selector) {
      const element = document.querySelector(deferredTarget.selector) as HTMLElement | null;
      if (element) return element;
    }
    return this.findHeadingElementByText(deferredTarget.headingText);
  };

  const attempt = (): boolean => {
    const element = findTarget();
    if (!element) return false;
    element.scrollIntoView({ behavior: 'smooth', block: 'center' });
    element.style.boxShadow = '0 0 0 3px rgba(67, 217, 200, 0.5)';
    element.style.borderRadius = '4px';
    setTimeout(() => {
      element.style.boxShadow = '';
    }, 2000);
    return true;
  };

  const start = () => {
    if (attempt()) return;
    const observer = new MutationObserver(() => {
      if (attempt()) observer.disconnect();
    });
    observer.observe(document.body, { childList: true, subtree: true });
    setTimeout(() => observer.disconnect(), 5000);
  };

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', start, { once: true });
  } else {
    start();
  }
}

// This method:
// 1. Retrieves the deferred target from sessionStorage
// 2. Cleans up by removing the item from sessionStorage
// 3. Tries to find the target element immediately
// 4. If not found, sets up a MutationObserver to watch for DOM changes
// 5. Times out after 5 seconds to prevent infinite waiting
// 6. Handles both DOMContentLoaded and already-loaded states
// The observer watches for any changes to the DOM that might introduce the target element

setupDeferredAction() & handleDeferredActions()

Deferred actions for cross-page navigation.

private setupDeferredAction(result: IndexRecord): void {
  const deferredAction = {
    selector: result.selector,
    type: result.type,
    label: result.label,
    destructive: result.destructive,
  };
  sessionStorage.setItem('reef-deferred-action', JSON.stringify(deferredAction));
  window.location.href = result.url;
}

handleDeferredActions(): void {
  const deferredActionStr = sessionStorage.getItem('reef-deferred-action');
  if (!deferredActionStr) return;

  try {
    const deferredAction = JSON.parse(deferredActionStr);
    sessionStorage.removeItem('reef-deferred-action');

    if (deferredAction.selector) {
      const element = document.querySelector(deferredAction.selector);
      if (element) {
        if (deferredAction.type === 'action' && !deferredAction.destructive) {
          const clickEvent = new MouseEvent('click', {
            bubbles: true,
            cancelable: true,
            view: window,
          });
          element.dispatchEvent(clickEvent);
        } else if (deferredAction.type === 'field') {
          (element as HTMLElement).focus();
        }
      }
    }
  } catch (error) {
    console.error('Failed to handle deferred action:', error);
    sessionStorage.removeItem('reef-deferred-action');
  }
}

// Deferred actions are used when:
// - User selects an action on a different page
// - The action is non-destructive (destructive actions are blocked)
// - We need to wait for the target page to load before executing
// The action is stored in sessionStorage and executed on page load
// Destructive actions are NOT executed automatically - they only navigate to the page

Safety Model

Reef implements a conservative safety model to prevent unintended actions:

Safety Rules

Scenario actionsMode = 'execute' actionsMode = 'navigate-only'
Non-destructive action on current page Executes the action Scrolls to element, no execution
Destructive action on current page Executes the action Scrolls to element, no execution
Action on different page Navigates to page, then executes (non-destructive only) Navigates to page, scrolls to element
Field on current page Focuses and selects the field Focuses and selects the field
Field on different page Navigates to page, then focuses Navigates to page, then focuses
Section navigation Scrolls to section Scrolls to section
Link/File/Media navigation Navigates to URL Navigates to URL

executeActionResult()

Handle action execution with safety checks.

private executeActionResult(result: IndexRecord, actionsMode?: string, showToast?: (msg: string) => void): void {
  if (result.destructive && actionsMode !== 'execute') {
    this.highlightAndNavigate(result);
    return;
  }

  const currentUrl = window.location.href.split('#')[0];
  const targetUrl = result.url.split('#')[0];

  if (currentUrl === targetUrl) {
    this.executeActionOnCurrentPage(result, showToast);
  } else {
    this.setupDeferredAction(result);
  }
}

// Key safety logic:
// 1. If action is destructive AND actionsMode is not 'execute':
//    - Only highlight and navigate, don't execute
// 2. If same page:
//    - Execute on current page
// 3. If different page:
//    - Set up deferred action (will execute on page load)
// Note: Destructive actions on different pages are still deferred and will execute
// unless actionsMode is 'navigate-only', in which case they won't be executed at all

// What is considered destructive?
// Actions with labels containing: 'delete', 'remove', 'pay', 'checkout', 'confirm', etc.
// See isDestructiveAction() in extraction.ts
// isDestructiveAction function from extraction.ts
function isDestructiveAction(name: string): boolean {
  const destructiveWords = [
    'delete', 'remove', 'erase', 'clear', 'destroy', 'kill',
    'pay', 'purchase', 'buy', 'checkout', 'order',
    'confirm', 'submit', 'send', 'publish',
    'logout', 'sign out', 'quit', 'exit',
    'cancel', 'stop', 'abort'
  ];
  const normalized = name.toLowerCase().trim();
  return destructiveWords.some(word => normalized.includes(word));
}

// Examples:
console.log(isDestructiveAction('Delete Account')); // true
console.log(isDestructiveAction('Save Changes'));   // false
console.log(isDestructiveAction('Checkout'));        // true
console.log(isDestructiveAction('Login'));           // false

Important Safety Notes:

  • Forms are never auto-filled or auto-submitted - Fields only receive focus, never automatic values
  • Destructive actions are flagged - But in 'execute' mode, they still run on the current page
  • Deferred actions respect actionsMode - Destructive deferred actions won't execute in 'navigate-only' mode
  • Cross-page actions are deferred - Action is stored and executed when the target page loads
  • Missing elements show toast - If element can't be found, user sees a notification

Recommendation: Use actionsMode='navigate-only' for production sites where you want to be extra safe.

Safe Action Execution

Implement custom safety logic for action execution.

import { ActionExecutor, isDestructiveAction } from './src/actions/index.js';

const executor = new ActionExecutor();

// Custom safe execute function
function safeExecute(result, showToast) {
  // Additional safety checks
  if (result.type === 'action' && result.destructive) {
    showToast('Destructive actions are disabled for safety');
    return;
  }
  
  if (result.selector?.includes('.admin')) {
    showToast('Admin actions require special permissions');
    return;
  }
  
  // Execute with standard safety
  executor.executeAction(result, 'execute', showToast);
}

// Usage
const result = { /* ... */ };
safeExecute(result, (msg) => console.log(msg));
