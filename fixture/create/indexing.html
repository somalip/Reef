Overview

The indexing module handles sitemap fetching, page crawling, and content extraction. The Indexer class is the main component that orchestrates the indexing process. 

/**
 * @file Indexing and crawling functionality for the search index.
 * Handles sitemap fetching, page crawling, and worker-based indexing.
 */

import {
  extractSections,
  extractActions,
  extractFields,
  extractLinks,
  extractFiles,
  extractMedia,
  extractStructuredData,
  extractHiddenContent,
} from '../extraction.js';
import {
  createSearchIndex,
  addToIndex,
  type SearchIndex,
  type IndexRecord,
  type TokenFilter,
} from '../search-index.js';
import type { ReefConfig } from '../types.js';

export class Indexer {
  private index: SearchIndex = createSearchIndex();
  private config: ReefConfig;

  constructor(config: ReefConfig) {
    this.config = config;
  }
  
  // ... methods
}

// Re-export from index.ts
export { Indexer } from './indexer.js';

// Usage
import { Indexer } from './src/indexing/index.js';
import type { ReefConfig } from './src/types.js';

const config: ReefConfig = { sitemap: '/sitemap.xml', maxPages: 100 };

The Indexer class:

  • Fetches and parses sitemap.xml files
  • Fetches HTML pages in parallel (with configurable concurrency)
  • Extracts content using the extraction module
  • Builds and maintains the search index
  • Handles caching via IndexedDB
  • Supports Web Worker offloading for large sites

Boot Process

The boot() method is the main entry point for indexing. It follows this sequence: 

async boot(onReady: () => void, showToast?: (msg: string) => void): Promise<void> {
  // 1. Try prebuilt index first if configured
  if (this.config.prebuiltIndexUrl) {
    try {
      const response = await fetch(this.config.prebuiltIndexUrl);
      if (response.ok) {
        const json = await response.text();
        this.index = deserializeIndex(json);
        console.info(`[reef] loaded prebuilt index with ${this.index.allSections.length} sections`);
        onReady();
        return;
      }
    } catch (error) {
      console.warn('[reef] prebuilt index fetch failed, falling back to crawling', error);
    }
  }

  // 2. Try IndexedDB cache first
  try {
    const { loadIndex } = await import('../cache.js');
    const ttl = this.config.ttl;
    const cached = await loadIndex(ttl);
    if (cached?.index && cached.index.allSections.length > 0) {
      this.index = cached.index;
      console.info(`[reef] loaded cached index with ${this.index.allSections.length} sections`);
      onReady();
      return;
    }
  } catch (error) {
    console.warn('[reef] cache load failed, continuing with fresh index', error);
  }

  // 3. Fetch and parse sitemap
  const candidates = this.getSitemapCandidates();
  
  for (const candidate of candidates) {
    try {
      const response = await fetch(candidate);
      if (!response.ok) continue;
      const xml = await response.text();
      const urls: string[] = [];
      const locRegex = /<loc>(.*?)<\/loc>/g;
      let match: RegExpExecArray | null;
      while ((match = locRegex.exec(xml)) !== null) {
        urls.push(match[1].trim());
      }

      const maxPages = this.config.maxPages ?? 500;
      if (this.config.useWorkerIndexing) {
        await this.fetchPagesWithWorker(urls.slice(0, maxPages), candidate, onReady);
      } else {
        const fetchedSections = await this.fetchPagesParallel(urls.slice(0, maxPages), candidate);
        if (fetchedSections.length) {
          addToIndex(this.index, fetchedSections, this.config.tokenizePipeline);
          console.info(`[reef] indexed ${fetchedSections.length} sections`);
        }
        
        // 4. Save to cache
        try {
          const { saveIndex } = await import('../cache.js');
          const versionHash = this.computeVersionHash(urls);
          const metadata = {
            versionHash,
            buildTime: Date.now(),
            pageMetadata: urls.reduce((acc, url) => ({ ...acc, [url]: versionHash }), {} as Record<string, string>)
          };
          await saveIndex(this.index, metadata);
        } catch (e) {
          console.warn('[reef] cache save failed', e);
        }
        
        onReady();
        return;
      }
    } catch (error) {
      console.warn('[reef] sitemap fetch failed', candidate, error);
    }
  }

  // 5. Fallback to same-origin crawl
  this.crawlSameOrigin(onReady);
}

Priority Order:

  1. Prebuilt Index - If prebuiltIndexUrl is configured, try to load prebuilt index first
  2. Cached Index - Try to load from IndexedDB cache if available and not expired
  3. Sitemap Crawling - Fetch and parse sitemap.xml, then crawl all pages
  4. Same-Origin Crawl - Fallback to crawling same-origin pages starting from current URL

Sitemap Crawling

getSitemapCandidates()

Get candidate sitemap URLs to try.

getSitemapCandidates(): string[] {
  const base = window.location.href;
  const configured = this.config.sitemap ?? 'sitemap.xml';
  const candidates = [this.resolveUrl(configured, base)];

  if (!configured.startsWith('/')) {
    candidates.push(this.resolveUrl('./sitemap.xml', base));
  } else if (configured === '/sitemap.xml') {
    candidates.push(this.resolveUrl('sitemap.xml', base));
  }

  return [...new Set(candidates)];
}

// Returns: ['https://example.com/custom-sitemap.xml', 'https://example.com/sitemap.xml']
// or: ['https://example.com/sitemap.xml', 'https://example.com/sitemap.xml'] (deduped)

fetchPagesParallel()

Fetch multiple pages in parallel with configurable concurrency.

private async fetchPagesParallel(urls: string[], sitemapUrl: string): Promise<IndexRecord[]> {
  const concurrency = 6; // Concurrent requests
  const sections: IndexRecord[] = [];
  const results: (IndexRecord[] | null)[] = new Array(urls.length);

  let idx = 0;
  const fetchBatch = async () => {
    while (idx < urls.length) {
      const i = idx++;
      const pageUrl = this.resolveUrl(urls[i], sitemapUrl);
      try {
        const pageResponse = await fetch(pageUrl);
        if (pageResponse.ok) {
          const html = await pageResponse.text();
          const pageSections = this.extractAllContent(html, pageUrl);
          results[i] = pageSections;
        }
      } catch {
        results[i] = null;
      }
    }
  };

  // Run 6 concurrent fetchers
  await Promise.all([...Array(concurrency)].map(() => fetchBatch()));

  for (let i = 0; i < results.length; i++) {
    if (results[i]) {
      sections.push(...(results[i] ?? []));
    }
  }

  return sections;
}

// Fetches up to 6 pages concurrently
// Each page is processed: HTML fetched -> content extracted -> records created
// Results are collected in order and returned as a flat array

fetchSitemapUrls()

Get all URLs from the sitemap without building the index.

async fetchSitemapUrls(): Promise<string[]> {
  const candidates = this.getSitemapCandidates();
  for (const candidate of candidates) {
    try {
      const response = await fetch(candidate);
      if (!response.ok) continue;
      const xml = await response.text();
      const urls: string[] = [];
      const locRegex = /<loc>(.*?)<\/loc>/g;
      let match: RegExpExecArray | null;
      while ((match = locRegex.exec(xml)) !== null) {
        urls.push(match[1].trim());
      }
      if (urls.length) return urls;
    } catch {
      continue;
    }
  }
  return [];
}

// Usage
window.Reef.getSitemapUrls().then(urls => {
  console.log('Found URLs:', urls);
  // ['/', '/about', '/contact', '/blog/post-1', ...]
});

Content Extraction

extractAllContent()

Extract all content types from HTML.

private extractAllContent(html: string, url: string): IndexRecord[] {
  const doc = new DOMParser().parseFromString(html, 'text/html');

  if (this.config.indexHidden) {
    extractHiddenContent(doc);
  }

  let rootElement: Element | Document = doc;
  if (this.config.scope) {
    const scopeElement = doc.querySelector(this.config.scope);
    if (scopeElement) {
      rootElement = scopeElement;
    }
  }

  const htmlToProcess = new XMLSerializer().serializeToString(rootElement as Element);

  const sections = extractSections(htmlToProcess, url);
  const actions = this.config.indexActions ? extractActions(htmlToProcess, url, this.config.excludeAction) : [];
  const fields = this.config.indexActions ? extractFields(htmlToProcess, url) : [];
  const links = extractLinks(htmlToProcess, url);
  const files = extractFiles(htmlToProcess, url, this.config.fileExtensions);
  const media = this.config.indexMedia ? extractMedia(htmlToProcess, url) : [];
  const structured = this.config.indexStructuredData ? extractStructuredData(htmlToProcess, url) : [];

  const combined: IndexRecord[] = [];
  combined.push(...sections, ...actions, ...fields, ...links, ...files, ...media, ...structured);
  return combined;
}

// This method:
// 1. Parses HTML into a DOM document
// 2. Optionally extracts hidden content (details, etc.)
// 3. Applies scope limitation if configured
// 4. Serializes back to HTML (to work with extraction functions)
// 5. Extracts all content types based on configuration
// 6. Combines all results into a single array

Configuration Options Affect Extraction:

  • indexActions - Controls whether actions and fields are extracted
  • indexMedia - Controls whether media elements are extracted
  • indexStructuredData - Controls whether JSON-LD structured data is extracted
  • indexHidden - Controls whether hidden content (details, etc.) is extracted
  • excludeAction - CSS selector for elements to exclude from action extraction
  • fileExtensions - File extensions to recognize for file link classification
  • scope - CSS selector to limit extraction to a specific part of the page

Worker Indexing

For large sites with many pages, Reef can offload HTML parsing to a Web Worker to prevent blocking the main thread.

fetchPagesWithWorker()

Fetch and index pages using a Web Worker.

private async fetchPagesWithWorker(
  urls: string[], 
  sitemapUrl: string, 
  onReady: () => void
): Promise<IndexRecord[]> {
  const workerUrl = new URL('../worker.js', import.meta.url).href;
  const worker = new Worker(workerUrl);

  return new Promise<IndexRecord[]>((resolve, reject) => {
    const messageHandler = async (e: MessageEvent) => {
      const { result, error, json } = e.data as { 
        result: string; 
        error: string; 
        json: string 
      };
      if (error) {
        console.error('[reef] worker error:', error);
        reject(new Error(error));
        return;
      }
      if (result === 'ok' && json) {
        worker.removeEventListener('message', messageHandler);
        worker.terminate();
        this.index = deserializeIndex(json);
        onReady();
        resolve([]);
      }
    };

    const errorHandler = (e: ErrorEvent) => {
      console.error('[reef] worker error:', e.error);
      reject(e.error);
    };

    worker.addEventListener('message', messageHandler);
    worker.addEventListener('error', errorHandler);

    // Fetch all HTML first
    const htmlMap: Record<string, string> = {};
    const fetchPromises = urls.map(async (pageUrl) => {
      try {
        const resolvedUrl = this.resolveUrl(pageUrl, sitemapUrl);
        const response = await fetch(resolvedUrl);
        if (response.ok) {
          htmlMap[resolvedUrl] = await response.text();
        }
      } catch (e) {
        console.warn('[reef] failed to fetch page for worker:', pageUrl);
      }
    });

    void Promise.all(fetchPromises).then(() => {
      const config = {
        scope: this.config.scope,
        indexActions: this.config.indexActions,
        indexMedia: this.config.indexMedia,
        indexStructuredData: this.config.indexStructuredData,
        indexHidden: this.config.indexHidden,
        excludeAction: this.config.excludeAction,
        fileExtensions: this.config.fileExtensions,
      };

      const id = Date.now();
      worker.postMessage({
        id,
        action: 'indexPages',
        payload: {
          pages: Object.entries(htmlMap),
          config,
        },
      });
    });
  });
}

// Usage: Enable via configuration
const config = {
  useWorkerIndexing: true,  // Enable Web Worker indexing
  maxPages: 500
};

const indexer = new Indexer(config);

How Worker Indexing Works:

  1. Main thread fetches all HTML content in parallel
  2. HTML content is passed to the Web Worker
  3. Worker parses HTML and extracts content using the same extraction functions
  4. Worker builds the search index
  5. Worker serializes the index to JSON and sends it back to the main thread
  6. Main thread deserializes the index and sets it as the current index

Benefits:

  • Prevents UI freezing during indexing of large sites
  • Allows main thread to remain responsive
  • Better performance for sites with hundreds of pages

Cache Integration

The Indexer automatically integrates with Reef's caching system to persist indexes between page loads.

Cache Loading

// In boot() method, after trying prebuilt index
try {
  const { loadIndex } = await import('../cache.js');
  const ttl = this.config.ttl; // Cache TTL in seconds
  const cached = await loadIndex(ttl);
  if (cached?.index && cached.index.allSections.length > 0) {
    this.index = cached.index;
    console.info(`[reef] loaded cached index with ${this.index.allSections.length} sections`);
    onReady();
    return;
  }
} catch (error) {
  console.warn('[reef] cache load failed, continuing with fresh index', error);
}

// Usage in ReefSearch
window.Reef.rebuildIndex().then(() => {
  // Index is loaded from cache or freshly built
  const records = window.Reef.getIndex();
  console.log('Records:', records.length);
});

Cache Saving

// In boot() method, after building fresh index
try {
  const { saveIndex } = await import('../cache.js');
  const versionHash = this.computeVersionHash(urls);
  const metadata = {
    versionHash,
    buildTime: Date.now(),
    pageMetadata: urls.reduce((acc, url) => ({ ...acc, [url]: versionHash }), {} as Record<string, string>)
  };
  await saveIndex(this.index, metadata);
} catch (e) {
  console.warn('[reef] cache save failed', e);
}

// Version hash is computed from all URLs
// This ensures cache invalidation when pages change
// pageMetadata stores a hash for each page for fine-grained invalidation

Cache Configuration

// Cache is controlled by these configuration options:
const config: ReefConfig = {
  // Cache TTL in seconds (default: 604800 = 7 days)
  ttl: 86400, // 1 day
  
  // Use Web Worker for indexing (helps with cache serialization performance)
  useWorkerIndexing: true,
  
  // Prebuilt index URL (alternative to cache)
  prebuiltIndexUrl: 'https://cdn.example.com/search-index.json'
};

// Force cache rebuild
window.Reef.clearCustomRecords();
window.Reef.rebuildIndex();

// Check cache status
const cached = await window.Reef.getSitemapUrls();
if (cached.length > 0) {
  // Cache likely has data
}

Examples

Custom Indexer

Create a custom Indexer with specific configuration.

import { Indexer } from './src/indexing/index.js';

const config = {
  sitemap: '/api/sitemap.xml',
  maxPages: 200,
  indexActions: true,
  indexMedia: false, // Skip media for performance
  indexStructuredData: true,
  indexHidden: false, // Skip hidden content
  excludeAction: '.admin-only, .internal',
  fileExtensions: 'pdf,docx,xlsx',
  useWorkerIndexing: true
};

const indexer = new Indexer(config);

// Start indexing
indexer.boot(() => {
  console.log('Indexing complete!');
  const index = indexer.getIndex();
  console.log('Total records:', index.allSections.length);
});

Same-Origin Crawl

Crawl pages on the same origin without a sitemap.

// If no sitemap is found, Reef falls back to same-origin crawling
// This starts from the current page and follows same-origin links

// Enable same-origin crawl
const config = {
  sitemap: '/nonexistent-sitemap.xml', // Will fail
  maxPages: 50
};

const indexer = new Indexer(config);

indexer.boot(() => {
  // Indexer will automatically fall back to same-origin crawl
  const index = indexer.getIndex();
  console.log('Crawled:', index.allSections.length, 'pages');
});

// The same-origin crawl:
// 1. Starts from window.location.href
// 2. Uses BFS to visit all same-origin links
// 3. Respects maxPages limit
// 4. Extracts content from each page
// 5. Indexes all found content

// Build a prebuilt index (in your build process)
import { createSearchIndex, addToIndex, serializeIndex } from './src/search.js';
import { extractSections } from './src/extraction.js';
import fs from 'fs';

const index = createSearchIndex();

// Add all your site's pages
const pages = getAllPages(); // Your function to get all HTML
pages.forEach(page => {
  const sections = extractSections(page.html, page.url);
  addToIndex(index, sections);
});

// Save to file
const json = serializeIndex(index);
fs.writeFileSync('./dist/search-index.json', json);
console.log('Prebuilt index saved!');

Scoped Indexing

Limit indexing to a specific part of each page.

// Only index content within #main-content
const config = {
  scope: '#main-content',
  sitemap: '/sitemap.xml',
  maxPages: 100
};

const indexer = new Indexer(config);

indexer.boot(() => {
  const index = indexer.getIndex();
  
  // All records will have content extracted from #main-content only
  // Navigation, headers, footers, etc. will be excluded
  console.log('Scoped index:', index.allSections.length, 'records');
});

// HTML example:
// <html>
//   <header>...</header>  <- Not indexed
//   <nav>...</nav>      <- Not indexed
//   <main id="main-content">
//     <h1>Page Title</h1>   <- Indexed
//     <p>Content...</p>      <- Indexed
//   </main>
//   <footer>...</footer>  <- Not indexed
// </html>

Custom Content Extraction

Customize what content is extracted and how.

// Configure what to index
const config = {
  sitemap: '/sitemap.xml',
  
  // Only index sections and links (skip actions, fields, media, files)
  indexActions: false,
  indexMedia: false,
  indexStructuredData: false,
  
  // Only index PDF files (skip other file types)
  fileExtensions: 'pdf',
  
  // Exclude certain actions
  excludeAction: '.no-index, .internal, .admin',
  
  // Don't index hidden content
  indexHidden: false
};

const indexer = new Indexer(config);

indexer.boot(() => {
  const index = indexer.getIndex();
  
  // Index will only contain sections and links
  const sections = index.allSections.filter(r => r.type === 'section');
  const links = index.allSections.filter(r => r.type === 'link');
  const files = index.allSections.filter(r => r.type === 'file');
  
  console.log('Sections:', sections.length);
  console.log('Links:', links.length);
  console.log('Files:', files.length);
});
