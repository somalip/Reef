readConfig()

The main method for reading configuration. Checks multiple sources in order.

static readConfig(): ReefConfig {
  // 1. Check for pending programmatic config
  if (this.pendingConfig) {
    const config = this.pendingConfig;
    this.pendingConfig = null;
    return this.mergeWithDefaults(config);
  }

  // 2. Check for script tag in DOM
  if (typeof document === 'undefined') {
    return this.getDefaultConfig();
  }

  const script = document.currentScript as HTMLScriptElement | null;
  const dataset = script?.dataset;

  if (!dataset) {
    return this.getDefaultConfig();
  }

  // 3. Parse dataset
  const datasetRecord: Record<string, string> = {};
  for (const key in dataset) {
    const value = dataset[key];
    if (value !== undefined) {
      datasetRecord[key] = value;
    }
  }

  return this.parseDataset(datasetRecord);
}

// Priority:
// 1. Pending programmatic config (from setConfig())
// 2. Script tag data attributes
// 3. Default config
// This allows programmatic config to override script tag config

parseDataset()

Parse configuration from a dataset object.

static parseDataset(dataset: Record<string, string>): ReefConfig {
  // Build tokenize pipeline if any text processing options are set
  let tokenizePipeline: TokenFilter[] | undefined = undefined;
  if (dataset.stemming === 'true' || dataset.stopwords === 'true' || dataset.diacritics === 'true') {
    const pipeline: TokenFilter[] = [];

    if (dataset.stopwords === 'true') {
      const stopWords = new Set([
        'a', 'an', 'the', 'and', 'or', 'but', 'in', 'on', 'at', 'to', 'for', 'of', 'with', 'by',
        'from', 'up', 'about', 'into', 'through', 'during', 'before', 'after', 'above', 'below',
      ]);
      pipeline.push((token: string) => stopWords.has(token) ? null : token);
    }

    if (dataset.diacritics === 'true') {
      pipeline.push((token: string) => token.normalize('NFD').replace(/[\u0300-\u036f]/g, ''));
    }

    if (dataset.stemming === 'true') {
      pipeline.push((token: string) => {
        if (token.length <= 2) return token;
        if (/ing$/.test(token)) return token.slice(0, -3);
        if (/ed$/.test(token)) return token.slice(0, -2);
        if (/es$/.test(token) && token.length > 3) return token.slice(0, -2);
        if (/s$/.test(token) && token.length > 3) return token.slice(0, -1);
        return token;
      });
    }

    tokenizePipeline = pipeline.length > 0 ? pipeline : undefined;
  }

  // Parse synonyms if provided
  let synonyms: Record<string, string[]> | undefined = undefined;
  if (dataset.synonyms) {
    try {
      synonyms = JSON.parse(dataset.synonyms);
    } catch {
      console.warn('[reef] Invalid synonyms JSON in data-synonyms attribute');
    }
  }

  // Return parsed config
  return {
    sitemap: dataset.sitemap ?? '/sitemap.xml',
    maxPages: Number(dataset.maxPages ?? 500),
    scope: dataset.scope,
    indexActions: dataset.indexActions !== 'false',
    indexMedia: dataset.indexMedia !== 'false',
    indexStructuredData: dataset.indexStructuredData !== 'false',
    indexHidden: dataset.indexHidden !== 'false',
    fileExtensions: dataset.fileExtensions,
    excludeAction: dataset.excludeAction,
    actionsMode: dataset.actionsMode as 'execute' | 'navigate-only' | undefined || 'execute',
    primaryColor: dataset.primaryColor,
    secondaryColor: dataset.secondaryColor,
    backgroundColor: dataset.backgroundColor,
    textColor: dataset.textColor,
    borderColor: dataset.borderColor,
    radius: dataset.radius ? Number(dataset.radius) : 24,
    theme: dataset.theme as 'light' | 'dark' | 'auto' | undefined,
    fontFamily: dataset.fontFamily,
    mode: dataset.mode ? dataset.mode as 'regular' | 'opaque' | 'high-contrast' : 'opaque',
    hotkey: dataset.hotkey,
    placeholder: dataset.placeholder,
    headless: dataset.headless === 'true',
    onReady: undefined,
    tokenizePipeline,
    synonyms,
    prebuiltIndexUrl: dataset.prebuiltIndexUrl,
    useWorkerIndexing: dataset.useWorkerIndexing === 'true',
    ttl: dataset.ttl ? Number(dataset.ttl) : undefined,
  };
}

// Dataset values are strings from data-* attributes
// Boolean values: 'true' or 'false' strings
// Numeric values: parsed with Number()
// JSON values: parsed with JSON.parse()
// Empty/undefined: use defaults

setConfig()

Set configuration programmatically before ReefSearch is instantiated.

import { ConfigReader } from './src/config/index.js';

// Set config before creating ReefSearch
ConfigReader.setConfig({
  sitemap: '/custom-sitemap.xml',
  maxPages: 100,
  actionsMode: 'navigate-only',
  primaryColor: '#8b5cf6',
  headless: true
});

// Then create ReefSearch (in ReefSearch constructor, readConfig() will use this)
import { ReefSearch } from './src/reef.js';
const reef = new ReefSearch();

// Or via window
window.Reef = reef;

// Note: setConfig() must be called BEFORE ReefSearch is instantiated
// Once ReefSearch is created, use its methods to change configuration

applyConfigToUI()

Apply configuration to the UI host element.

static applyConfigToUI(host: HTMLDivElement, config: ReefConfig & { mode?: string }): void {
  const cfg = config;
  host.style.setProperty('--primary-color', cfg.primaryColor ?? '#66d9c8');
  host.style.setProperty('--secondary-color', cfg.secondaryColor ?? '#ff8562');
  host.style.setProperty('--background-color', cfg.backgroundColor ?? 'rgba(20,30,28,0.88)');
  host.style.setProperty('--text-color', cfg.textColor ?? '#edebe6');
  host.style.setProperty('--border-color', cfg.borderColor ?? 'rgba(255,255,255,0.1)');
  host.style.setProperty('--radius', cfg.radius?.toString() ?? '24');
  host.style.setProperty('--font-family', cfg.fontFamily ?? 'Inter, system-ui, sans-serif');

  host.classList.remove('mode-regular', 'mode-opaque', 'mode-high-contrast');
  switch ((cfg.mode ?? 'opaque') as string) {
    case 'opaque':
      host.classList.add('mode-opaque');
      break;
    case 'high-contrast':
      host.classList.add('mode-high-contrast');
      break;
    default:
      host.classList.add('mode-regular');
  }
}

// This method:
// 1. Sets CSS custom properties (variables) on the host element
// 2. Applies the appropriate mode class
// 3. Uses defaults if values are not provided
// 4. Cleans up previous mode classes before applying new one

// CSS Variables Applied:
// - --primary-color: Primary accent color
// - --secondary-color: Secondary accent color
// - --background-color: Modal background color
// - --text-color: Text color
// - --border-color: Border color
// - --radius: Border radius in pixels
// - --font-family: Font family for the UI
/* Example CSS using config variables */
:host {
  --primary-color: #66d9c8;
  --secondary-color: #ff8562;
  --background-color: rgba(20,30,28,0.88);
  --text-color: #edebe6;
  --border-color: rgba(255,255,255,0.1);
  --radius: 24px;
  --font-family: 'Inter', system-ui, sans-serif;
}

.panel {
  background: var(--background-color);
  color: var(--text-color);
  border: 1px solid var(--border-color);
  border-radius: var(--radius);
  font-family: var(--font-family);
}

.icon {
  stroke: var(--primary-color);
}

/* Mode-specific styles */

  background: rgba(20,30,28,0.92) !important;
}

Parsing Logic

Boolean Parsing

How boolean values are parsed from script attributes.

// From parseDataset()
indexActions: dataset.indexActions !== 'false',
// If data-index-actions="false", result is false
// If data-index-actions="true" or any other value, result is true
// If data-index-actions is not set, result is true (default)

// Examples:
console.log('true' !== 'false');  // true
console.log('false' !== 'false'); // false
console.log('yes' !== 'false');   // true
console.log('1' !== 'false');      // true
console.log('' !== 'false');       // true
console.log(undefined !== 'false'); // true

// This means: any value EXCEPT 'false' is treated as true
// To disable: data-index-actions="false"
// To enable: data-index-actions="true" (or omit it)

Numeric Parsing

How numeric values are parsed.

// From parseDataset()
maxPages: Number(dataset.maxPages ?? 500),
// If data-max-pages is set, parse it as a number
// If not set or invalid, use default (500)

// Examples:
console.log(Number('100'));     // 100
console.log(Number('50'));      // 50
console.log(Number('abc'));     // NaN
console.log(Number(''));        // 0
console.log(Number(undefined)); // NaN

// With fallback:
const value = dataset.maxPages ?? 500;
const result = Number(value);
// If value is 'abc', result is NaN
// But Number('abc') is still used, which becomes 0 or NaN
// In practice, this works because invalid numbers fall back to defaults

// Better approach (not currently in code, but could be):
const parseNumber = (value, defaultValue) => {
  const num = Number(value);
  return Number.isFinite(num) ? num : defaultValue;
};

console.log(parseNumber('100', 500)); // 100
console.log(parseNumber('abc', 500)); // 500
console.log(parseNumber('', 500));    // 500

JSON Parsing

How JSON values (like synonyms) are parsed.

// From parseDataset()
let synonyms: Record<string, string[]> | undefined = undefined;
if (dataset.synonyms) {
  try {
    synonyms = JSON.parse(dataset.synonyms);
  } catch {

// Example script tag:
<script src="reef.min.js"
  data-synonyms='{"install": ["setup", "configuration"], "help": ["support", "docs"]}'
></script>

// Result:
// synonyms = {
//   install: ['setup', 'configuration'],
//   help: ['support', 'docs']
// }

// Invalid JSON example:
<script src="reef.min.js"
  data-synonyms='{invalid json}'
></script>
// Result: synonyms = undefined, warning logged to console

Token Filter Pipeline

How text processing pipeline is built from configuration.

// From parseDataset()
let tokenizePipeline: TokenFilter[] | undefined = undefined;
if (dataset.stemming === 'true' || dataset.stopwords === 'true' || dataset.diacritics === 'true') {
  const pipeline: TokenFilter[] = [];

  if (dataset.stopwords === 'true') {
    const stopWords = new Set([/* ... */]);
    pipeline.push((token: string) => stopWords.has(token) ? null : token);
  }

  if (dataset.diacritics === 'true') {
    pipeline.push((token: string) => token.normalize('NFD').replace(/[\u0300-\u036f]/g, ''));
  }

  if (dataset.stemming === 'true') {
    pipeline.push((token: string) => {
      if (token.length <= 2) return token;
      if (/ing$/.test(token)) return token.slice(0, -3);
      if (/ed$/.test(token)) return token.slice(0, -2);
      if (/es$/.test(token) && token.length > 3) return token.slice(0, -2);
      if (/s$/.test(token) && token.length > 3) return token.slice(0, -1);
      return token;
    });
  }

  tokenizePipeline = pipeline.length > 0 ? pipeline : undefined;
}

// Enable via script attributes:
<script src="reef.min.js"
  data-stopwords="true"
  data-stemming="true"
  data-diacritics="true"
></script>

// Or programmatically:
ConfigReader.setConfig({
  tokenizePipeline: [
    (token) => stopWords.has(token) ? null : token,
    (token) => token.normalize('NFD').replace(/[\u0300-\u036f]/g, '')
  ]
});
