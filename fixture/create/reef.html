Overview

ReefSearch is the main class that orchestrates all of Reef's functionality. It manages the UI, indexing, search, actions, and provides a comprehensive API for developers.

Constructor

new ReefSearch(config?: Partial<ReefConfig>)

Creates a new ReefSearch instance.

config Partial<ReefConfig> Optional

Partial configuration object. Any provided values override defaults.

// Default constructor
const reef = new ReefSearch();

// With configuration
const reef = new ReefSearch({
  sitemap: '/custom-sitemap.xml',
  maxPages: 100,
  headless: true,
  actionsMode: 'navigate-only'
});

Note: When using the script tag approach, ReefSearch is automatically instantiated and assigned to window.Reef. You don't need to call the constructor manually.

Core Methods

open()

Opens the search modal.

// Open the search modal
window.Reef.open();

// With custom placeholder
window.Reef.setPlaceholder('Search documentation...');
window.Reef.open();

close()

Closes the search modal.

// Close the search modal
window.Reef.close();

openWithQuery(query: string)

Opens the search modal with a pre-filled query.

query string

The search query to pre-fill.

// Open with pre-filled query
window.Reef.openWithQuery('installation');

isOpenState(): boolean

Returns whether the search modal is currently open.

if (window.Reef.isOpenState()) {
  console.log('Search modal is open');
}

UI Methods

setHeadless(headless: boolean): void

Switch between headless and regular mode.

// Switch to headless mode
window.Reef.setHeadless(true);

// Switch back to regular mode (requires re-initialization)
window.Reef.setHeadless(false);

setColorScheme(scheme: object): void

Update the color scheme of the UI.

window.Reef.setColorScheme({
  primary: '#ff6b6b',
  secondary: '#4ecdc4',
  background: 'rgba(255,255,255,0.8)',
  text: '#111111',
  border: '#cccccc',
  radius: 12
});

setTheme(theme: 'light' | 'dark' | 'auto'): void

Set the theme mode.

// Available themes
window.Reef.setTheme('light');
window.Reef.setTheme('dark');
window.Reef.setTheme('auto'); // Auto-detect system preference

setMode(mode: 'regular' | 'opaque' | 'high-contrast'): void

Set the UI mode.

// Available modes
window.Reef.setMode('regular');      // Standard appearance
window.Reef.setMode('opaque');       // More opaque background
window.Reef.setMode('high-contrast'); // High contrast for accessibility

setHotkey(hotkey: string): void

Set custom hotkey for opening the search modal.

// Single hotkey
window.Reef.setHotkey('altk'); // Opens with Alt+K

// Multiple hotkeys
window.Reef.setHotkey('ctrlk,cmdk'); // Opens with Ctrl+K or Cmd+K
window.Reef.setHotkey('ctrlshiftk,f'); // Opens with Ctrl+Shift+K or Ctrl+F

setPlaceholder(placeholder: string): void

Set custom placeholder text for the search input.

window.Reef.setPlaceholder('Search docs...');

setFontFamily(fontFamily: string): void

Set custom font family for the UI.

window.Reef.setFontFamily('system-ui, sans-serif');
window.Reef.setFontFamily('\'Inter\', -apple-system, BlinkMacSystemFont, sans-serif');

getHotkey(): string

Get the current hotkey configuration.

const currentHotkey = window.Reef.getHotkey();
// Returns: "ctrlk,cmdk" by default

onselect(callback: (result: IndexRecord) => void): void

Register a callback for when a search result is selected.

window.Reef.onselect(function(result) {
  console.log('Selected:', result.type, result.headingText);
  // result properties:
  // - type: 'section' | 'action' | 'field' | 'link' | 'file' | 'media' | 'structured'
  // - headingText: string
  // - url: string
  // - breadcrumb: string
  // - bodyText: string
  // - destructive: boolean (for actions)
  // - selector: string (for actions/fields)
});

offselect(): void

Remove the selection callback.

// Remove the callback
window.Reef.offselect();

Search Methods

search(query: string, limit?: number): IndexRecord[]

Search the index programmatically.

// Search for "installation"
const results = window.Reef.search('installation', 10);

// Search with default limit
const results = window.Reef.search('getting started');

searchSections(query: string, options?: SearchOptions | number): ScoredRecord[]

Advanced search with options.

// With options object
const results = window.Reef.searchSections('installation', {
  limit: 10,
  fuzzy: true,
  fuzzyThreshold: 0.6,
  includeScore: true,
  includeMatches: true
});

// With limit number (shorthand)
const results = window.Reef.searchSections('installation', 10);

getIndex(): IndexRecord[]

Get all indexed records.

// Get all indexed records
const allRecords = window.Reef.getIndex();

// Filter by type
const sections = allRecords.filter(r => r.type === 'section');
const actions = allRecords.filter(r => r.type === 'action');
const fields = allRecords.filter(r => r.type === 'field');

addCustomRecords(records: IndexRecord[]): void

Add custom records to the index.

window.Reef.addCustomRecords([{
  id: 'custom-1',
  url: window.location.href,
  headingText: 'Custom Result',
  headingId: 'custom-1',
  breadcrumb: 'Custom',
  bodyText: 'Custom searchable content',
  type: 'section'
}, {
  id: 'custom-action',
  url: window.location.href,
  headingText: 'My Custom Action',
  headingId: 'custom-action',
  breadcrumb: 'Custom',
  bodyText: 'Custom action description',
  type: 'action',
  selector: '#my-button',
  destructive: false
}]);

clearCustomRecords(): void

Clear all custom records from the index.

// Clear custom records
window.Reef.clearCustomRecords();

reindex(): void

Rebuild the index from scratch (re-crawls sitemap).

// Rebuild the index
window.Reef.reindex();

rebuildIndex(): Promise<void>

Rebuild the index and return a promise that resolves when complete.

// Rebuild and wait for completion
window.Reef.rebuildIndex().then(() => {
  console.log('Index rebuilt');
  const records = window.Reef.getIndex();
  console.log('Total records:', records.length);
});

getInteractiveRecords(): IndexRecord[]

Get all interactive records (actions and fields).

// Get all interactive elements
const interactive = window.Reef.getInteractiveRecords();

// Filter by type
const actions = interactive.filter(r => r.type === 'action');
const fields = interactive.filter(r => r.type === 'field');

getAgentTools(): Array<{ name, description, type, selector?, id }>

Get all interactive records formatted for AI agent tool use.

// Get agent tools
const tools = window.Reef.getAgentTools();

// Each tool has:
// - name: string
// - description: string
// - type: 'action' | 'field'
// - selector: string | undefined
// - id: string

console.log('Available tools:', tools.length);

// Find a specific tool
const loginButton = tools.find(t => t.name.includes('Login'));

act(recordId: string): Promise<{ success: boolean; reason?: string }>

Execute an action by record ID.

// Execute an action by ID
window.Reef.act('page.html#action-0').then(result => {
  if (result.success) {
    console.log('Action executed successfully');
  } else {
    console.log('Action failed:', result.reason);
    // Possible reasons: 'not-found', 'blocked-destructive', 'error', 'element-not-found'
  }
});

fillField(recordId: string, value: string): Promise<{ success: boolean; reason?: string }>

Fill a field with a value programmatically.

// Fill a field by ID
window.Reef.fillField('page.html#field-2', 'user@example.com').then(result => {
  if (result.success) {
    console.log('Field filled successfully');
  } else {
    console.log('Fill failed:', result.reason);
    // Possible reasons: 'not-found', 'not-a-field', 'no-selector', 'element-not-found', 'error'
  }
});

suggest(query: string, limit?: number): string[]

Get query suggestions based on the index.

// Get suggestions
const suggestions = window.Reef.suggest('inst', 5);
// Returns: ['installation', 'instructions', 'instance', ...]

facets(): Record<string, number>

Get facet counts for indexed content.

// Get facet counts
const facets = window.Reef.facets();
// Returns: { 'getting': 5, 'started': 3, 'installation': 2, ... }

trackQuery(query: string): void

Track a query for popularity analytics.

// Track a user query
window.Reef.trackQuery('installation');

getPopularQueries(n?: number): string[]

Get the most popular queries.

// Get top 10 popular queries
const popular = window.Reef.getPopularQueries(10);
// Returns: ['installation', 'getting started', 'api', ...]

getSitemapUrls(): Promise<string[]>

Get all URLs from the sitemap without building the index.

// Get sitemap URLs
window.Reef.getSitemapUrls().then(urls => {
  console.log('Found URLs:', urls);
});

Agent Methods

agent(): Agent

Get a new Agent instance for programmatic automation.

// Get an agent
const agent = window.Reef.agent();

// Chain actions
await agent
  .click('#login-button')
  .type('#email', 'user@example.com')
  .type('#password', 'secret')
  .submit();

executeWorkflow(definition: WorkflowStep[] | object, options?: WorkflowOptions): Promise<void>

Execute a multi-step workflow.

// Array of steps
await window.Reef.executeWorkflow([
  { action: 'navigate', url: '/dashboard' },
  { action: 'click', selector: '#refresh' },
  { action: 'wait', timeout: 1000 },
  { action: 'extract', selector: '#data-table' }
]);

// Object with steps and options
await window.Reef.executeWorkflow({
  steps: [
    { action: 'click', selector: '#login' },
    { action: 'type', selector: '#email', value: 'test@example.com' }
  ],
  options: {
    maxRetries: 3,
    retryDelay: 500,
    stopOnError: true,
    onStepError: (step, index, error) => {
      console.error(`Step ${index} failed:`, error);
    }
  }
});

toggleInspector(force?: boolean): void

Toggle the visual inspector for debugging.

// Toggle inspector
window.Reef.toggleInspector();

// Force enable
window.Reef.toggleInspector(true);

// Force disable
window.Reef.toggleInspector(false);

Configuration Methods

getConfig(): ReefConfig

Get the current configuration.

const config = window.Reef.getConfig();
console.log(config.hotkey, config.mode, config.maxPages);
// Output: "ctrlk,cmdk" "opaque" 500

setOnReady(callback: (data: { index: IndexRecord[] }) => void): void

Set a callback that fires when the index is ready.

// Set onReady callback
window.Reef.setOnReady(({ index }) => {
  console.log('Index ready with', index.length, 'records');
  
  // Now you can safely search
  const results = window.Reef.search('installation');
});

// Check if already ready
if (window.Reef.getIndex().length > 0) {
  // Index is already ready
}

Examples

Basic Usage

Simple site search with default configuration.

<script src="reef.min.js"></script>

Headless Mode

Use Reef for programmatic search without UI.

// Initialize
window.Reef = new ReefSearch({ headless: true });

// Set callback when ready
window.Reef.setOnReady(({ index }) => {
  // Search programmatically
  const results = window.Reef.search('installation');
  console.log('Results:', results);
});

Customize the appearance of the search modal.

window.Reef.setColorScheme({

Agent Automation

Use the Agent API for automation workflows.

const agent = window.Reef.agent();

// Define a workflow
const loginWorkflow = async () => {
  await agent
    .click('#login-button')
    .type('#email', 'user@example.com')
    .type('#password', 'secret123')
    .submit();
  
  // Wait for redirect
  await agent.wait(2000);
  
  // Verify we're on the dashboard
  const title = await agent.extract('h1');
  console.log('Page title:', title);
};

// Execute
loginWorkflow().catch(console.error);

Result Selection Callback

Handle user selection of search results.

window.Reef.onselect(function(result) {
  // Log the selected result
  console.log('User selected:', {
    type: result.type,
    title: result.headingText,
    url: result.url
  });
  
  // Custom analytics
  trackEvent('search_select', {
    result_type: result.type,
    query: result.headingText
  });
});
