Overview

This document contains all TypeScript type definitions for the Reef library. These types are defined in src/types.ts and are exported for use in your projects.

Importing Types

Import types from the main types file:

// Import specific types
import type { 
  IndexRecord, 
  ReefConfig, 
  WorkflowStep, 
  AgentSession 
} from './src/types.js';

// Or import all types
import type * as ReefTypes from './src/types.js';

// Use in your code
const record: IndexRecord = {
  id: 'test-1',
  url: 'https://example.com',
  headingText: 'Test Section',
  headingId: 'test-1',
  breadcrumb: 'Test',
  bodyText: 'Test content',
  type: 'section'
};

const config: ReefConfig = {
  sitemap: '/sitemap.xml',
  maxPages: 500,
  actionsMode: 'execute'
};

Index records are the core data structures that represent searchable content in Reef. They are created during the indexing process and stored in the search index.

SectionDocument

Base interface for all document types.

export interface SectionDocument {
  id: string;              // Unique identifier for the document
  url: string;            // URL of the page containing this section
  headingText: string;    // The heading or title text
  headingId: string;      // The HTML ID of the heading element
  breadcrumb: string;     // Breadcrumb path for context
  bodyText: string;      // The full text content
}

Used by: All index records extend from SectionDocument.

IndexRecord

The main record type that extends SectionDocument with additional metadata.

export interface IndexRecord extends SectionDocument {
  type: 'section' | 'action' | 'field' | 'link' | 'file' | 'media' | 'structured';
  selector?: string;      // CSS selector for the element
  destructive?: boolean;  // Whether the action is destructive
  label?: string;         // Alternative label text
  value?: string;         // Value for form fields
  transcript?: string;    // Transcript for media
  structuredData?: any;   // Structured data (JSON-LD, etc.)
}

Record Types

Type Description Behavior
'section' Page sections and headings Navigates to the section anchor
'action' Clickable elements (buttons, links) Clicks the element or navigates first if on different page
'field' Form fields (inputs, textareas, selects) Focuses and selects the field
'link' External links Navigates to the external URL
'file' Downloadable files Navigates to the file resource
'media' Images, videos, audio Navigates to the page containing the media
'structured' Structured data (JSON-LD, FAQ, etc.) Shows inline preview and navigates to source 
// Example of different record types
const sectionRecord: IndexRecord = {
  id: 'page.html#introduction',

  headingText: 'Introduction',

  headingId: 'submit-button',
  breadcrumb: 'Form',

const fieldRecord: IndexRecord = {

Configuration Types

ReefConfig

The main configuration interface for ReefSearch.

export interface ReefConfig {
  // Sitemap and crawling
  sitemap?: string;              // Sitemap path (default: '/sitemap.xml')
  maxPages?: number;            // Maximum pages to fetch (default: 500)
  scope?: string;               // CSS selector for scope limitation
  
  // Indexing options
  indexActions?: boolean;       // Index buttons/actions (default: true)
  indexMedia?: boolean;         // Index images, video, audio (default: true)
  indexStructuredData?: boolean; // Index JSON-LD structured data (default: true)
  indexHidden?: boolean;        // Index hidden content (default: true)
  
  // File handling
  fileExtensions?: string;      // Comma-separated file extensions
  excludeAction?: string;       // CSS selectors to exclude from actions
  
  // Actions
  actionsMode?: 'execute' | 'navigate-only'; // (default: 'execute')
  
  // UI styling
  primaryColor?: string;        // Primary accent color
  secondaryColor?: string;      // Secondary accent color
  backgroundColor?: string;    // Modal background color
  textColor?: string;           // Text color
  borderColor?: string;         // Border color
  radius?: number;              // Border radius in pixels (default: 24)
  
  // Theme and appearance
  theme?: 'light' | 'dark' | 'auto'; // Theme mode
  fontFamily?: string;          // Font family
  mode?: 'regular' | 'opaque' | 'high-contrast'; // UI mode
  
  // Hotkey
  hotkey?: string;              // Keyboard shortcut (default: 'ctrlk,cmdk')
  placeholder?: string;         // Input placeholder text
  
  // Mode
  headless?: boolean;           // Headless mode (default: false)
  
  // Callbacks
  onReady?: (data: { index: IndexRecord[] }) => void;
  
  // Tokenization
  tokenizePipeline?: TokenFilter[];
  synonyms?: Record<string, string[]>;
  
  // Cache
  prebuiltIndexUrl?: string;    // URL for prebuilt index JSON
  useWorkerIndexing?: boolean;  // Use Web Worker for indexing (default: false)
  ttl?: number;                 // Cache TTL in seconds
}

TokenFilter

Function type for text tokenization preprocessing.

export type TokenFilter = (token: string) => string | null;

// Example token filters
const stopWordsFilter: TokenFilter = (token) => {
  const stopWords = new Set(['the', 'a', 'an', 'and', 'or', 'but']);
  return stopWords.has(token.toLowerCase()) ? null : token;
};

const stemmingFilter: TokenFilter = (token) => {
  if (token.endsWith('ing')) return token.slice(0, -3);
  if (token.endsWith('ed')) return token.slice(0, -2);
  return token;
};

// Usage in configuration
const config: Partial<ReefConfig> = {
  tokenizePipeline: [stopWordsFilter, stemmingFilter]
};

Search Types

SearchOptions

Options for the searchSections function.

export interface SearchOptions {
  limit?: number;              // Maximum number of results (default: 8)
  fuzzy?: boolean;             // Enable fuzzy search (default: false)
  fuzzyThreshold?: number;     // Fuzzy match threshold (0-1)
  fuzzyDistance?: number;      // Maximum edit distance for fuzzy matching
  includeScore?: boolean;      // Include score in results
  includeMatches?: boolean;    // Include match positions
  weights?: Partial<Record<'headingText' | 'bodyText' | 'label' | 'breadcrumb', number>>;
  extended?: boolean;          // Enable extended query syntax
  scoringAlgorithm?: 'reef-classic' | 'bm25'; // Scoring algorithm
  filter?: (record: IndexRecord) => boolean;  // Filter function for results
  sortFn?: (a: ScoredRecord, b: ScoredRecord) => number; // Custom sort function
  typeWeights?: Partial<Record<IndexRecord['type'], number>>; // Weight by type
}
// Example search options
const options: SearchOptions = {
  limit: 20,
  fuzzy: true,
  fuzzyThreshold: 0.7,
  includeScore: true,
  weights: {
    headingText: 3.0,
    bodyText: 1.0,
    label: 2.0,
    breadcrumb: 0.5
  },
  filter: (record) => record.type === 'section' || record.type === 'action',
  sortFn: (a, b) => b.score - a.score
};

// Usage
const results = searchSections(query, index, options);

MatchSpan

Represents a matched span in the text.

export interface MatchSpan {
  key: string;        // The key (field name) where the match was found
  start: number;      // Start position of the match
  end: number;        // End position of the match
}

ScoredRecord

A search result with score and match information.

export interface ScoredRecord {
  record: IndexRecord;   // The matched record
  score: number;        // The relevance score
  matches?: MatchSpan[]; // Array of match positions
}

// Example usage
const results: ScoredRecord[] = searchSections(query, index, {
  includeScore: true,
  includeMatches: true
});

results.forEach(result => {
  console.log(`Score: ${result.score}`);
  console.log(`Record: ${result.record.headingText}`);
  if (result.matches) {
    console.log(`Matches: ${result.matches.length}`);
  }
});

Workflow Types

WorkflowStep

A single step in a workflow.

export interface WorkflowStep {
  action: 'click' | 'type' | 'navigate' | 'extract' | 'submit' | 'back' | 'forward' | 'wait';
  selector?: string;    // CSS selector for the target element
  value?: string;       // Value to type (for type action)
  url?: string;         // URL to navigate to (for navigate action)
  recordId?: string;    // IndexRecord ID as alternative to selector
  timeout?: number;    // Timeout in milliseconds (for wait action)
}

// Example workflow steps
const steps: WorkflowStep[] = [
  { action: 'navigate', url: '/login' },
  { action: 'type', selector: '#email', value: 'user@example.com' },
  { action: 'type', selector: '#password', value: 'secret' },
  { action: 'click', selector: '#submit' },
  { action: 'wait', timeout: 2000 },
  { action: 'extract', selector: 'h1' }
];

WorkflowOptions

Configuration options for workflow execution.

export interface WorkflowOptions {
  maxRetries?: number;            // Maximum retries per step (default: 0)
  retryDelay?: number;            // Delay between retries in ms (default: 500)
  stopOnError?: boolean;          // Stop on first error (default: true)
  onStepStart?: (step: WorkflowStep, index: number) => void;
  onStepComplete?: (step: WorkflowStep, index: number, result: any) => void;
  onStepError?: (step: WorkflowStep, index: number, error: Error) => void;
}

// Example workflow options
const options: WorkflowOptions = {
  maxRetries: 3,
  retryDelay: 1000,
  stopOnError: false,
  onStepStart: (step, index) => {
    console.log(`Starting step ${index}: ${step.action}`);
  },
  onStepComplete: (step, index, result) => {
    console.log(`Completed step ${index}: ${step.action}`);
  },
  onStepError: (step, index, error) => {
    console.error(`Error in step ${index}: ${step.action}`, error);
  }
};

// Usage
await agent.executeWorkflow(steps, options);

Agent Types

InspectorInterface

Interface for the visual inspector component.

export type InspectorInterface = {
  activate: () => void;          // Activate the inspector
  deactivate: () => void;        // Deactivate the inspector
  isActive: () => boolean;       // Check if inspector is active
  setRecords: (records: IndexRecord[]) => void; // Set records to inspect

ActionResult

Result of an action execution.

export interface ActionResult {
  success: boolean;        // Whether the action succeeded
  reason?: string;         // Reason for failure (if any)
  element?: Element;      // The element that was acted upon
}

// Example usage
const result: ActionResult = await agent.click('#button');
if (result.success) {
  console.log('Action succeeded');
} else {
  console.log('Action failed:', result.reason);
}

AgentSession

Session information for the agent.

export interface AgentSession {
  id: string;              // Unique session identifier
  url: string;            // Current page URL
  timestamp: number;      // Session creation timestamp
  cookies?: Record<string, string>;     // Current browser cookies
  localStorage?: Record<string, string>; // Current localStorage data
}

// Example usage
const session: AgentSession = agent.getSession();
console.log('Session ID:', session.id);
console.log('Current URL:', session.url);
console.log('Cookies:', session.cookies);
