Overview

The worker.ts module provides a Web Worker that runs expensive indexing and search operations in a background thread, preventing UI blocking.

Background Processing

Runs CPU-intensive indexing off the main thread

Non-Blocking UI

Prevents UI freezing during large index operations

Index Management

Create, serialize, and search indexes in worker

Page Extraction

Extract content from HTML pages in background

Message Format

Request Message

Messages sent to the worker must include an action and optional payload.

interface WorkerRequest {
  id: number;          // Request identifier for matching responses
  action: string;     // Action to perform
  payload?: any;      // Action-specific data
}

// Example request
const request: WorkerRequest = {
  id: 1,
  action: 'search',
  payload: {
    query: 'my search term',
    options: { limit: 10 }
  }
};

// Send to worker
worker.postMessage(request);

Response Message

Worker responses include the request ID for correlation.

interface WorkerResponse {
  id: number;          // Matches request ID
  result?: 'ok';      // Success indicator
  results?: any;      // Action-specific results
  json?: string;      // Serialized index data
  error?: string;     // Error message if any
}

// Example response (search)
const searchResponse: WorkerResponse = {
  id: 1,
  results: [
    { id: 'page#section-1', headingText: 'Results', ... },
    { id: 'page#section-2', headingText: 'More Results', ... }
  ]
};

// Example response (indexPages)
const indexResponse: WorkerResponse = {
  id: 2,
  result: 'ok',
  json: '{"allSections": [...], ...}'  // Serialized index
};

// Example error response
const errorResponse: WorkerResponse = {
  id: 3,
  error: 'Unknown action'
};

Available Actions

Action Payload Response Description
createIndex none { result: 'ok' } Create a new empty search index
search { query, options? } { results: IndexRecord[] } Search the index with a query
deserializeIndex { json: string } { result: 'ok' } Load a serialized index from JSON
serializeIndex none { json: string } Serialize the current index to JSON
indexPages { pages, config } { result: 'ok', json: string } Extract and index content from HTML pages

indexPages Payload

The indexPages action accepts pages as HTML content and configuration.

interface IndexPagesPayload {
  pages: Array<[url: string, html: string]>;
  config: {
    scope?: string;              // CSS selector to scope extraction
    indexActions?: boolean;      // Extract interactive elements
    indexMedia?: boolean;        // Extract media elements
    indexStructuredData?: boolean; // Extract structured data
    indexHidden?: boolean;       // Extract hidden content
    excludeAction?: string;      // CSS selector to exclude actions
    fileExtensions?: string;    // Comma-separated file extensions
  };
}

// Example
const payload: IndexPagesPayload = {
  pages: [
    ['https://example.com', '<html>...</html>'],
    ['https://example.com/about', '<html>...</html>']
  ],
  config: {
    indexActions: true,
    indexMedia: true,
    indexStructuredData: true,
    fileExtensions: 'pdf,doc,docx'
  }
};

Source Code 

/// <reference lib="webworker" />
import {
  createSearchIndex,
  searchSections,
  deserializeIndex,
  serializeIndex,
  addToIndex,
  type SearchIndex,
  type IndexRecord,
} from './search-index.js';
import {
  extractSections,
  extractActions,
  extractFields,
  extractLinks,
  extractFiles,
  extractMedia,
  extractStructuredData,
  extractHiddenContent,
} from './extraction.js';

let index: SearchIndex = createSearchIndex();

self.onmessage = (e: MessageEvent) => {
  const { id, action, payload } = e.data as { id: number; action: string; payload: any };
  
  switch (action) {
    case 'createIndex':
      index = createSearchIndex();
      self.postMessage({ id, result: 'ok' });
      break;
      
    case 'search': {
      const { query, options } = payload as { query: string; options?: any };
      const results = searchSections(query, index, options);
      self.postMessage({ id, results });
      break;
    }
    
    case 'deserializeIndex': {
      index = deserializeIndex(payload.json);
      self.postMessage({ id, result: 'ok' });
      break;
    }
    
    case 'serializeIndex': {
      const json = serializeIndex(index);
      self.postMessage({ id, json });
      break;
    }
    
    case 'indexPages': {
      (async () => {
        const { pages, config } = payload as { 
          pages: [string, string][]; 
          config: { 
            scope?: string; 
            indexActions?: boolean; 
            indexMedia?: boolean; 
            indexStructuredData?: boolean; 
            indexHidden?: boolean; 
            excludeAction?: string; 
            fileExtensions?: string 
          } 
        };
        const allRecords: IndexRecord[] = [];
        
        for (const [pageUrl, html] of pages) {
          try {
            const doc = new DOMParser().parseFromString(html, 'text/html');
            
            if (config.indexHidden) {
              extractHiddenContent(doc);
            }
            
            let rootElement: Element | Document = doc;
            if (config.scope) {
              const scopeElement = doc.querySelector(config.scope);
              if (scopeElement) {
                rootElement = scopeElement;
              }
            }
            
            const htmlToProcess = new XMLSerializer().serializeToString(rootElement as Element);
            const sections = extractSections(htmlToProcess, pageUrl);
            const actions = config.indexActions ? extractActions(htmlToProcess, pageUrl, config.excludeAction) : [];
            const fields = config.indexActions ? extractFields(htmlToProcess, pageUrl) : [];
            const links = extractLinks(htmlToProcess, pageUrl);
            const files = extractFiles(htmlToProcess, pageUrl, config.fileExtensions);
            const media = config.indexMedia ? extractMedia(htmlToProcess, pageUrl) : [];
            const structured = config.indexStructuredData ? extractStructuredData(htmlToProcess, pageUrl) : [];
            
            allRecords.push(...sections, ...actions, ...fields, ...links, ...files, ...media, ...structured);
          } catch (error) {
            console.error('[reef-worker] error indexing page:', pageUrl, error);
          }
        }
        
        addToIndex(index, allRecords);
        const json = serializeIndex(index);
        self.postMessage({ id, result: 'ok', json });
      })();
      break;
    }
    default:
      self.postMessage({ id, error: 'Unknown action' });
  }
};

Usage

Creating a Worker

Create and communicate with the Reef worker.

// Create worker from source
const worker = new Worker('./src/worker.js', { type: 'module' });

// Or from bundled distribution
const worker = new Worker('./dist/worker.js', { type: 'module' });

// Track request IDs
let requestId = 0;

// Helper function to send messages
function sendToWorker(action, payload = {}) {
  const id = ++requestId;
  return new Promise((resolve, reject) => {
    const listener = (e) => {
      if (e.data.id === id) {
        worker.removeEventListener('message', listener);
        if (e.data.error) {
          reject(new Error(e.data.error));
        } else {
          resolve(e.data);
        }
      }
    };
    worker.addEventListener('message', listener);
    worker.postMessage({ id, action, payload });
  });
}

Using the Worker for Indexing

Offload indexing to the worker thread.

// Create worker
const worker = new Worker('./src/worker.js', { type: 'module' });

// Create a new index in the worker
async function createIndex() {
  await sendToWorker('createIndex');
  console.log('Index created');
}

// Index multiple pages
async function indexPages(pages) {
  const result = await sendToWorker('indexPages', {
    pages: pages.map(p => [p.url, p.html]),
    config: {
      indexActions: true,
      indexMedia: true,
      indexStructuredData: true
    }
  });
  
  // result.json contains the serialized index
  const indexJson = result.json;
  localStorage.setItem('reef-index', indexJson);
  
  return indexJson;
}

// Search in the worker
async function search(query) {
  const result = await sendToWorker('search', {
    query: query,
    options: { limit: 10, fuzzy: true }
  });
  
  return result.results; // Array of IndexRecord
}

// Load saved index
async function loadIndex(json) {
  await sendToWorker('deserializeIndex', { json });
}

// Save current index
async function saveIndex() {
  const result = await sendToWorker('serializeIndex');
  return result.json; // Serialized index
}

Complete Example

Full worker usage example with error handling.

class ReefWorker {
  constructor(workerUrl) {
    this.worker = new Worker(workerUrl, { type: 'module' });
    this.requestId = 0;
    this.pendingRequests = new Map();
  }
  
  send(action, payload = {}) {
    const id = ++this.requestId;
    return new Promise((resolve, reject) => {
      const listener = (e) => {
        if (e.data.id === id) {
          this.worker.removeEventListener('message', listener);
          if (e.data.error) {
            reject(new Error(e.data.error));
          } else {
            resolve(e.data);
          }
        }
      };
      this.worker.addEventListener('message', listener);
      this.worker.postMessage({ id, action, payload });
    });
  }
  
  async indexSite(pages) {
    try {
      // Create fresh index
      await this.send('createIndex');
      
      // Index all pages
      const result = await this.send('indexPages', {
        pages,
        config: { indexActions: true, indexMedia: true }
      });
      
      return result.json; // Get serialized index
    } catch (error) {
      console.error('Indexing failed:', error);
      throw error;
    }
  }
  
  async search(query, options) {
    const result = await this.send('search', { query, options });
    return result.results;
  }
  
  terminate() {
    this.worker.terminate();
  }
}

// Usage
const reefWorker = new ReefWorker('./src/worker.js');

// Index pages in background
await reefWorker.indexSite([
  ['/', document.documentElement.outerHTML],
  ['/about', aboutPageHtml]
]);

// Search
await reefWorker.search('my query');
