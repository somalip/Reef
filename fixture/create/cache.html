openDB()

Opens the Reef IndexedDB database. Creates the database and object store if they don't exist.

export async function openDB(): Promise<IDBDatabase> {
  return new Promise((resolve, reject) => {
    const req = indexedDB.open(DB_NAME, 1);
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
    req.onupgradeneeded = () => {
      const db = req.result;
      if (!db.objectStoreNames.contains(STORE_NAME)) {
        db.createObjectStore(STORE_NAME, { keyPath: CACHE_VERSION_KEY });
      }
    };
  });
}

// Constants
const DB_NAME = 'reef-index';
const STORE_NAME = 'indices';
const CACHE_VERSION_KEY = 'version';

// Usage
const db = await openDB();
console.log('Database opened:', db.name);

saveIndex()

Saves a search index with metadata to the database.

export async function saveIndex(
  index: SearchIndex, 
  metadata: CacheMetadata
): Promise<void> {
  const db = await openDB();
  const tx = db.transaction(STORE_NAME, 'readwrite');
  const store = tx.objectStore(STORE_NAME);
  
  const data = {
    [CACHE_VERSION_KEY]: metadata.versionHash,
    index: serializeIndex(index),
    metadata
  };
  
  await new Promise<void>((resolve, reject) => {
    const req = store.put(data);
    req.onsuccess = () => resolve();
    req.onerror = () => reject(req.error);
  });
}

import { saveIndex } from './src/cache.js';
import { createSearchIndex, addToIndex } from './src/search-index.js';

const index = createSearchIndex();

  versionHash: 'abc123',
  buildTime: Date.now(),
  pageCount: 10,
  configHash: 'xyz789'
};

await saveIndex(index, metadata);

loadIndex()

Loads a cached index from the database. Optionally checks TTL expiration.

export async function loadIndex(ttl?: number): Promise<
  { index: SearchIndex; metadata: CacheMetadata } | null
> {
  const db = await openDB();
  const tx = db.transaction(STORE_NAME, 'readonly');
  const store = tx.objectStore(STORE_NAME);
  
  const allRecords = await new Promise<any[]>((resolve, reject) => {
    const req = store.getAll();
    req.onsuccess = () => resolve(req.result);
    req.onerror = () => reject(req.error);
  });

  if (!allRecords.length) return null;
  
  const record = allRecords[0] as { [key: string]: any };
  const cached = deserializeIndex(record.index);
  const metadata: CacheMetadata = record.metadata;

  if (ttl && metadata.buildTime) {
    const age = Date.now() - metadata.buildTime;
    if (age > ttl) {
      return null; // Expired
    }
  }

  return { index: cached, metadata };
}

// Usage
// Load without TTL check
const { index, metadata } = await loadIndex();

// Load with TTL check (expire after 1 hour)
const freshIndex = await loadIndex(1000 * 60 * 60);

// Check if loaded
if (freshIndex) {
  console.log('Index loaded with', freshIndex.index.allSections.length, 'sections');
} else {
  console.log('No cached index or expired');
}

const STORE_NAME = 'indices';
const CACHE_VERSION_KEY = 'version';

Usage

Basic Usage

Saving and loading indexes with TTL-based expiration.

import { saveIndex, loadIndex, clearCache } from './src/cache.js';
import { createSearchIndex, addToIndex } from './src/search-index.js';

// Create and populate index
const index = createSearchIndex();
await addToIndex(index, records);

// Save to cache with metadata
const metadata = {
  versionHash: 'v1-abc123',
  buildTime: Date.now(),
  pageCount: records.length,
  configHash: 'default-config'
};

await saveIndex(index, metadata);
console.log('Index saved to cache');

// Load from cache (no TTL check)
const cached = await loadIndex();
if (cached) {
  console.log('Loaded index with', cached.index.allSections.length, 'sections');
  console.log('Metadata:', cached.metadata);
}

// Load with TTL check (1 hour expiration)
const freshIndex = await loadIndex(1000 * 60 * 60);
if (freshIndex) {
  // Index is fresh, use it
} else {
  // Index expired or doesn't exist, rebuild
  await rebuildAndSaveIndex();
}

Integration with ReefSearch

How cache is used in the main ReefSearch class.

// In ReefSearch class (reef.ts)
import { saveIndex, loadIndex } from './cache.js';

class ReefSearch {
  async init() {
    // Try to load cached index
    const cached = await loadIndex(this.config.cacheTTL);
    
    if (cached && cached.metadata.versionHash === this.getVersionHash()) {
      this.index = cached.index;
      this.diagnostics.cachedIndex = true;
    } else {
      // Build new index
      await this.buildIndex();
      
      // Save to cache
      await saveIndex(this.index, {
        versionHash: this.getVersionHash(),
        buildTime: Date.now(),
        pageCount: this.index.allSections.length,
        configHash: this.config.hash
      });
    }
  }
  
  async rebuildIndex() {
    // Clear old cache
    await clearCache();
    
    // Build new index
    await this.buildIndex();
    
    // Save fresh
    await saveIndex(this.index, {
      versionHash: this.getVersionHash(),
      buildTime: Date.now(),
      pageCount: this.index.allSections.length,
      configHash: this.config.hash
    });
  }
}

TTL-Based Cache Invalidation

Automatically expire old indexes based on time-to-live.

// Configuration
const CACHE_TTL = {
  // Expire after 1 hour
  default: 1000 * 60 * 60,
  
  // Expire after 15 minutes for development
  development: 1000 * 60 * 15,
  
  // Expire after 1 day for production
  production: 1000 * 60 * 60 * 24,
  
  // Never expire
  persistent: 0
};

// Usage
const config = {
  cacheTTL: CACHE_TTL.production
};

// Load with TTL check
await loadIndex(config.cacheTTL);

// Version-based invalidation
// If config changes, versionHash changes, cache is ignored
const metadata = {
  versionHash: computeHashFromConfig(config), // Different on config change
  buildTime: Date.now(),
  ...
};

// Force cache refresh
async function refreshCache() {
  await clearCache();
  // Rebuild index
}

Error Handling

Handling IndexedDB errors and edge cases.

import { saveIndex, loadIndex, clearCache } from './src/cache.js';

// Wrap cache operations in try-catch
async function safeSaveIndex(index, metadata) {
  try {
    await saveIndex(index, metadata);
    return true;
  } catch (error) {
    console.error('Failed to save index:', error);
    // Fall back to in-memory only
    return false;
  }
}

async function safeLoadIndex(ttl) {
  try {
    return await loadIndex(ttl);
  } catch (error) {
    console.error('Failed to load index:', error);
    return null; // Return null to trigger rebuild
  }
}

// Check IndexedDB support
function isIndexedDBSupported() {
  return typeof indexedDB !== 'undefined';
}

// Polyfill for browsers without IndexedDB
if (!isIndexedDBSupported()) {
  // Use in-memory cache only
  console.warn('IndexedDB not available, using in-memory cache');
} {
  if (estimate.quota < 100 * 1024 * 1024) {
    console.warn('Low storage quota, cache may be limited');
  }
});
