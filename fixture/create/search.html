Overview

The search.ts module is the main export point for search-related functionality. It re-exports functions from both search-index.ts and extraction.ts. 

/**
 * @file Public API exports for search module.
 * Re-exports functions from search-index.ts and extraction.ts for external use.
 */

export type { 
  IndexRecord, 
  SectionDocument, 
  SearchOptions, 
  ScoredRecord, 
  MatchSpan, 
  TokenFilter 
} from './types.js';

export {
  createSearchIndex,

  addSectionsToIndex,
  getAllSections,

  findClosestWord,

The search module provides:

  • Index Management: Create, update, serialize, and deserialize search indexes
  • Querying: Search, suggest, facet, and track queries
  • Extraction: Extract sections, actions, fields, links, files, and media from HTML
  • Text Processing: Tokenization, stemming, and text normalization

Search Index

The search-index module provides the core indexing and querying functionality. For detailed documentation, see the Search Index documentation.

Basic Usage

import { 
  createSearchIndex, 
  addToIndex, 
  searchSections 
} from './src/search.js';

// Create an index
const index = createSearchIndex();

// Add records
const records = [
  {
    id: 'page1#section1',
    url: '/page1',
    headingText: 'Introduction',
    headingId: 'section1',
    breadcrumb: 'Getting Started',
    bodyText: 'Welcome to our application...',
    type: 'section'
  },
  {
    id: 'page1#login-button',
    url: '/page1',
    headingText: 'Login',
    headingId: 'login-button',
    breadcrumb: 'Auth',
    bodyText: 'Click to login',
    type: 'action',
    selector: '#login-button',
    destructive: false
  }
];

addToIndex(index, records);

// Search the index
const results = searchSections('login', index, 10);
console.log('Results:', results);

Serialization

import { 
  createSearchIndex, 
  addToIndex, 
  serializeIndex, 
  deserializeIndex 
} from './src/search.js';

// Create and populate index
const index = createSearchIndex();
addToIndex(index, records);

// Serialize to JSON
const json = serializeIndex(index);
console.log('Serialized:', json.substring(0, 100) + '...');

// Deserialize from JSON
const restoredIndex = deserializeIndex(json);
console.log('Restored records:', restoredIndex.allSections.length);

// Save to localStorage
localStorage.setItem('reef-index', serializeIndex(index));

// Load from localStorage
const saved = localStorage.getItem('reef-index');
if (saved) {
  const index = deserializeIndex(saved);
}

Query Suggestions

import { createSearchIndex, addToIndex, suggest } from './src/search.js';

const index = createSearchIndex();
addToIndex(index, records);

// Get suggestions
const suggestions = suggest('inst', index, 5);
console.log('Suggestions:', suggestions);
// Output: ['installation', 'instructions', 'instance', ...]

// Get facets (term frequencies)
import { facets } from './src/search.js';
const termCounts = facets(index);
console.log('Term counts:', termCounts);
// Output: { 'welcome': 1, 'application': 1, 'login': 1, ... }

// Track queries for popularity
import { trackQuery, getPopularQueries } from './src/search.js';
trackQuery(index, 'installation');
trackQuery(index, 'getting started');
trackQuery(index, 'installation'); // Duplicate

const popular = getPopularQueries(index, 3);
console.log('Popular queries:', popular);
// Output: ['installation', 'getting started'] (sorted by frequency)

Extraction

The extraction module provides functions for extracting various types of content from HTML. For detailed documentation, see the Extraction documentation.

Basic Extraction

import { extractSections } from './src/search.js';

const html = `
<html>
<body>
  <h1 id="intro">Introduction</h1>
  <p>Welcome to our application.</p>
  <h2 id="features">Features</h2>
  <p>Our app has many features.</p>
</body>
</html>
`;

const sections = extractSections(html, 'https://example.com');
console.log('Sections:', sections);
// Output: Array of IndexRecord objects for each section

// Extract actions (buttons, links)
import { extractActions } from './src/search.js';
const actions = extractActions(html, 'https://example.com');
console.log('Actions:', actions);

// Extract fields (form inputs)
import { extractFields } from './src/search.js';
const fields = extractFields(html, 'https://example.com');
console.log('Fields:', fields);

// Extract links
import { extractLinks } from './src/search.js';
const links = extractLinks(html, 'https://example.com');
console.log('Links:', links);

// Extract files
import { extractFiles } from './src/search.js';
const files = extractFiles(html, 'https://example.com', 'pdf,doc,docx');
console.log('Files:', files);

// Extract media
import { extractMedia } from './src/search.js';
const media = extractMedia(html, 'https://example.com');
console.log('Media:', media);

// Extract structured data (JSON-LD)
import { extractStructuredData } from './src/search.js';
const structured = extractStructuredData(html, 'https://example.com');
console.log('Structured data:', structured);

Utility Functions

import { 
  stripTags, 
  generateSelector, 
  extractHeadingId, 
  resolveUrl 
} from './src/search.js';

// Strip HTML tags
const cleanText = stripTags('<p>Hello <strong>World</strong></p>');
console.log(cleanText); // 'Hello World'

// Generate CSS selector
const element = document.querySelector('#main-content');
const selector = generateSelector(element);
console.log(selector); // '#main-content'

// Extract heading ID
const heading = document.querySelector('h1');
const headingId = extractHeadingId(heading);
console.log(headingId); // The ID of the h1 element

// Resolve URL
const absolute = resolveUrl('/page.html', 'https://example.com/base/');
console.log(absolute); // 'https://example.com/base/page.html'

const absolute2 = resolveUrl('page.html', 'https://example.com/base/');
console.log(absolute2); // 'https://example.com/base/page.html'

const absolute3 = resolveUrl('https://other.com', 'https://example.com/base/');
console.log(absolute3); // 'https://other.com'

Examples

Build a Custom Search Index

Create and manage a custom search index.

import { 
  createSearchIndex, 
  addToIndex, 
  searchSections 
} from './src/search.js';

// Create index
const index = createSearchIndex();

// Add custom content
const customRecords = [
  {
    id: 'faq-1',
    url: '/faq#installation',
    headingText: 'How to Install',
    headingId: 'faq-1',
    breadcrumb: 'FAQ',
    bodyText: 'To install Reef, add the script tag...',
    type: 'section'
  },
  {
    id: 'faq-2',
    url: '/faq#configuration',
    headingText: 'Configuration Options',
    headingId: 'faq-2',
    breadcrumb: 'FAQ',
    bodyText: 'Reef supports various configuration options...',
    type: 'section'
  }
];

addToIndex(index, customRecords);

// Search
const results = searchSections('install', index, 5);
console.log('FAQ Results:', results);

Extract from Multiple Pages

Extract content from multiple HTML pages.

import { 
  createSearchIndex, 
  addToIndex, 
  extractSections, 
  extractActions, 
  extractFields 
} from './src/search.js';

const index = createSearchIndex();

// Array of pages to index
const pages = [
  { url: '/', html: '<html>...</html>' },
  { url: '/about', html: '<html>...</html>' },
  { url: '/contact', html: '<html>...</html>' }
];

// Extract and index all pages
pages.forEach(page => {
  const sections = extractSections(page.html, page.url);
  const actions = extractActions(page.html, page.url);
  const fields = extractFields(page.html, page.url);
  
  addToIndex(index, [...sections, ...actions, ...fields]);
});

console.log('Total indexed:', index.allSections.length);

Advanced Search with Options

Use advanced search options for better results.

import { 
  createSearchIndex, 
  addToIndex, 
  searchSections 
} from './src/search.js';

const index = createSearchIndex();
addToIndex(index, records);

// Search with fuzzy matching
const results = searchSections('instalation', index, {
  fuzzy: true,
  fuzzyThreshold: 0.7,
  limit: 10
});

// Search with custom weights
const weightedResults = searchSections('search', index, {
  weights: {
    headingText: 3.0,
    bodyText: 1.0,
    label: 2.0
  },
  includeScore: true
});

// Search with filtering
const filteredResults = searchSections('button', index, {
  filter: record => record.type === 'action',
  limit: 5
});

// Search with type weights
const typedResults = searchSections('data', index, {
  typeWeights: {
    action: 2.0,
    field: 1.5,
    section: 1.0
  }
});

Save and Load Index

Persist index to localStorage or IndexedDB.

import { 
  createSearchIndex, 
  addToIndex, 
  serializeIndex, 
  deserializeIndex 
} from './src/search.js';

// Create and populate index
const index = createSearchIndex();
addToIndex(index, records);

// Save to localStorage
const json = serializeIndex(index);
localStorage.setItem('reef-index-backup', json);

// Later, load from localStorage
const savedJson = localStorage.getItem('reef-index-backup');
if (savedJson) {
  const restoredIndex = deserializeIndex(savedJson);
  console.log('Restored:', restoredIndex.allSections.length, 'records');
}

// For IndexedDB (using Reef's cache)
import { saveIndex, loadIndex } from './src/cache.js';
const cached = await loadIndex();
if (cached) {
  const index = cached.index;
  console.log('Loaded from cache:', index.allSections.length);
}

// Save to cache
await saveIndex(index, {
  versionHash: 'v1',
  buildTime: Date.now(),
  pageMetadata: {}
});

Query Suggestions

Provide query suggestions to users.

import { 
  createSearchIndex, 
  addToIndex, 
  suggest, 
  trackQuery, 
  getPopularQueries 
} from './src/search.js';

const index = createSearchIndex();
addToIndex(index, records);

// Track user queries
const userQueries = ['install', 'setup', 'getting', 'config'];
userQueries.forEach(query => {
  trackQuery(index, query);
});

// Get suggestions for partial query
const input = 'inst';
const suggestions = suggest(input, index, 5);
console.log('Suggestions:', suggestions);

// Get popular queries
const popular = getPopularQueries(index, 3);
console.log('Popular:', popular);

// Autocomplete function
function getAutocompleteSuggestions(input, limit = 5) {
  return suggest(input, index, limit);
}

// Usage
const autocomplete = getAutocompleteSuggestions('set');
console.log('Autocomplete:', autocomplete);

Extract Structured Data

Extract JSON-LD and other structured data.

import { extractStructuredData } from './src/search.js';

const htmlWithStructuredData = `
<html>
<head>
  <script type="application/ld+json">
  {
    "@context": "https://schema.org",
    "@type": "Person",
    "name": "John Doe",
    "jobTitle": "Developer",
    "worksFor": {
      "@type": "Organization",
      "name": "Example Inc"
    }
  }
  </script>
</head>
<body>
  <h1>John Doe</h1>
</body>
</html>
`;

const structured = extractStructuredData(htmlWithStructuredData, 'https://example.com');
console.log('Structured data:', structured);

// structured will contain IndexRecord objects with structuredData property
structured.forEach(record => {
  if (record.structuredData) {
    console.log('Found structured data:', record.structuredData);
    // Output: The parsed JSON-LD object
  }
});
