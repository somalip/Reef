Quick Start

Basic Installation

Copy this to your project to get started:

<!-- Method 1: Local file -->
<script src="reef.min.js"></script>

<!-- Method 2: CDN (jsDelivr) -->
<script src="https://cdn.jsdelivr.net/npm/reef-search@latest/dist/reef.min.js"></script>

<!-- Method 3: With configuration -->
<script src="reef.min.js" 
        data-sitemap="/sitemap.xml"
        data-max-pages="500"
        data-hotkey="ctrlk,cmdk"
        data-placeholder="Search this site..."></script>
// NPM Package
npm install reef-search

// ESM Import
import { ReefSearch } from 'reef-search';
const reef = new ReefSearch();

// Global access
window.Reef = reef;

Configuration Options

Sitemap Settings

data-sitemap="/sitemap.xml" - Path to your sitemap

data-max-pages="500" - Max pages to index

data-scope="main" - Limit indexing to CSS selector

Content Indexing

data-index-actions="true" - Index buttons/toggles

data-index-fields="true" - Index form inputs

data-index-media="true" - Index images/video

data-index-hidden="true" - Include collapsed content

UI Settings

data-theme="dark|light|auto" - Theme mode

data-mode="regular|opaque|high-contrast" - Display mode

data-hotkey="ctrlk,cmdk" - Open shortcut

data-placeholder="Search..." - Input placeholder

Actions Mode

data-actions-mode="execute" - Auto-click actions

data-actions-mode="navigate-only" - Navigate only (safer)

Core API Methods

Modal Control

// Open/close the search modal
window.Reef.open();
window.Reef.close();

// Open with a pre-filled query
window.Reef.openWithQuery('installation');

// Check if modal is open
if (window.Reef.isOpenState()) {
  console.log('Search is open');
}

// Get current configuration
const config = window.Reef.getConfig();
console.log('Hotkey:', config.hotkey);
// Selection callback
window.Reef.onselect((result) => {
  console.log('Selected:', result.headingText, result.url);
  // Custom analytics
  gtag('event', 'search_select', {
    term: result.headingText,
    url: result.url
  });
});

// Remove callback
window.Reef.offselect();

Agentic API (Chainable Actions)

Agent Methods

Use window.Reef.agent() for programmatic DOM interaction:

// Chainable API
const agent = window.Reef.agent();

// Click elements
await agent.click('#myButton');

// Type into inputs
await agent.type('#emailField', 'user@example.com');

// Submit forms
await agent.submit('form');

// Navigate
await agent.navigate('/page.html');

// Wait for async operations
await agent.wait(1000);

// Extract content
const text = await agent.extract('#element');

Record Types

section section

Page headings and body text. Navigate to heading anchor.

action action

Clickable elements (buttons, toggles). Execute via search.

field field

Form inputs, textareas, selects. Focus from search.

link link

External links. Navigate to different origin.

file file

Downloadable files (PDF, DOC, etc.).

media media

Images with alt text, video/audio with transcripts.

structured structured

FAQ entries, JSON-LD structured data.

Headless Mode

Disable UI for Custom Implementation

<!-- Method 1: Data attribute -->
<script src="reef.min.js" data-headless="true"></script>

<!-- Method 2: Runtime -->
window.Reef.setHeadless(true);
// Access index programmatically
window.Reef.setOnReady(({ index }) => {
  console.log('Index ready:', index.length, 'records');
});

// Search without UI
const results = window.Reef.search('query', 10);

// Get all records
const allRecords = window.Reef.getIndex();

// Custom search UI
function renderCustomResults(query) {
  const results = window.Reef.search(query, 20);
  const container = document.getElementById('custom-results');
  container.innerHTML = results.map(r => 
    `<div class="result">
      <strong>${r.headingText}</strong>
      <span>${r.type}</span>
    </div>`
  ).join('');
}

Parity Features (Fuse.js, MiniSearch, Pagefind)

Extended Query Syntax

Try these queries in search: "exact phrase", install NOT beta, install OR setup

// Extended syntax supported:
// "exact phrase" - exact match
// term1 OR term2 - OR operator  
// term1 NOT term2 - exclude term
// term* - wildcard prefix

const results = window.Reef.searchSections('install OR setup', {
  scoringAlgorithm: 'bm25',
  includeScore: true
});

Advanced Search Options

Weighted Scoring

// Custom weights - prioritize headings over body text
const results = window.Reef.searchSections('installation', {
  weights: { headingText: 100, bodyText: 30 },
  includeScore: true
});

Agent Actions

// Get agent instance
const agent = window.Reef.agent();

// Click an element
await agent.click('#myButton');

// Type into a field

// Submit a form
await agent.submit('form');

// Navigate to a page
await agent.navigate('/settings');

// Go back/forward in history
await agent.back();
await agent.forward();

// Wait for async operations
await agent.wait(1000);

// Extract text content
const text = await agent.extract('#content');

Execute Workflows

// Define a workflow
const workflow = [
  { action: 'type', selector: '#email', value: 'user@example.com' },
  { action: 'type', selector: '#password', value: 'secret' },
  { action: 'click', selector: '#loginBtn' }
];

// Execute with options
await window.Reef.executeWorkflow(workflow, {
  maxRetries: 2,
  retryDelay: 500,
  stopOnError: true,
  onStepStart: (step, i) => console.log(`Step ${i}: ${step.action}`),
  onStepComplete: (step, i) => console.log(`Completed step ${i}`),
  onStepError: (step, i, err) => console.error(`Error at step ${i}:`, err)
});

act() and fillField()

Fuzzy Matching

// Fuzzy search with custom threshold

  fuzzy: true,
  fuzzyThreshold: 2,
  fuzzyDistance: 100,
  includeMatches: true
});

// Get "did you mean" suggestions

// Check matches
results.forEach(r => {
  console.log(`${r.record.headingText}: ${r.score}`);
  r.matches?.forEach(m => {
    console.log(`  ${m.key}: [${m.start}-${m.end}]`);
  });
});
